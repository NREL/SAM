# download wxWidgets 3.1.0 to ~
bunzip2 -d wxWidgets-3.1.0.tar.bz2
tar xvzf wxWidgets-3.1.0.tar
cd wxWidgets-3.1.0
# edit wxconfigure.linux64 script below to use your home directory rather than
# 'adobos'

make install
