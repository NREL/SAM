on_load{'Financial CS Costs'} = define()
{
	property( 'cs_cost_recurring_fixed','FixedLength',${analysis_period});
	property( 'cs_cost_recurring_capacity','FixedLength',${analysis_period});
	property( 'cs_cost_recurring_generation','FixedLength',${analysis_period});
	enable_cost_escalation_rate();
};

on_change{'cs_cost_recurring_fixed'} = define()
{
	enable('cs_cost_recurring_fixed_escal', #${cs_cost_recurring_fixed} == 1 ); 
};

on_change{'cs_cost_recurring_capacity'} = define()
{
	enable('cs_cost_recurring_capacity_escal', #${cs_cost_recurring_capacity} == 1 ); 
};

on_change{'cs_cost_recurring_generation'} = define()
{
	enable('cs_cost_recurring_generation_escal', #${cs_cost_recurring_generation} == 1 ); 
};

function enable_cost_escalation_rate() 
{
	enable('cs_cost_recurring_fixed_escal', #${cs_cost_recurring_fixed} == 1 ); 
	enable('cs_cost_recurring_capacity_escal', #${cs_cost_recurring_capacity} == 1 ); 
	enable('cs_cost_recurring_generation_escal', #${cs_cost_recurring_generation} == 1 ); 
}
