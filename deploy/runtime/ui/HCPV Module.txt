equations{ 'hcpv.module.rad4X' } = define() {
	return ${module_concentration}*${hcpv.module.rad4}*0.001; };

equations{ 'hcpv.module.area' } = define() {
	return ${module_concentration}*${module_cell_area}*0.0001*${module_ncells}; };

equations{ 'hcpv.module.cell_temp' } = define() {
	return ?${module_reference} [  ${hcpv.module.rad0},  ${hcpv.module.rad1},  ${hcpv.module.rad2},  ${hcpv.module.rad3},  ${hcpv.module.rad4} ] 
	*  ( exp(${module_a} 
	     + 4*${module_b} ) 
	       + ${module_dT}/1000.0) + 20 ;
	};

equations{ 'hcpv.module.rad2X' } = define() {
	return ${module_concentration}*${hcpv.module.rad2}*0.001; };

equations{ 'hcpv.module.rad0X' } = define() {
	return ${module_concentration}*${hcpv.module.rad0}*0.001; };

equations{ 'hcpv.module.power' } = define() {
	return ?${module_reference}
[
 ${hcpv.module.mjeff0}/100*${hcpv.module.rad0},
 ${hcpv.module.mjeff1}/100*${hcpv.module.rad1},
 ${hcpv.module.mjeff2}/100*${hcpv.module.rad2},
 ${hcpv.module.mjeff3}/100*${hcpv.module.rad3},
 ${hcpv.module.mjeff4}/100*${hcpv.module.rad4}
]
 * ${hcpv.module.area} 
 * ${module_optical_error} 
 * (1 - ${module_flutter_loss_coeff}*4) /* reference wind speed */
 * ${module_alignment_error}
 * ${hcpv.module.mam_ref}; };

equations{ 'hcpv.module.rad3X' } = define() {
	return ${module_concentration}*${hcpv.module.rad3}*0.001; };

equations{ 'hcpv.module.est_eff' } = define() {
	return ?${module_reference}
[
 ${hcpv.module.mjeff0},
 ${hcpv.module.mjeff1},
 ${hcpv.module.mjeff2},
 ${hcpv.module.mjeff3},
 ${hcpv.module.mjeff4}
]
 * ${module_optical_error} 
 * (1 - ${module_flutter_loss_coeff}*4) /* reference wind speed */
 * ${module_alignment_error}
 * ${hcpv.module.mam_ref}; };

equations{ 'hcpv.module.rad1X' } = define() {
	return ${module_concentration}*${hcpv.module.rad1}*0.001; };

equations{ 'hcpv.module.mam_ref' } = define() {
	return ${module_a0}
	     + ${module_a1}*1.5 
	     + ${module_a2}*2.25 
	     + ${module_a3}*5.0625 
	     + ${module_a4}*7.59375;
};

equations{ 'module_mjeff' } = define() {
	arr[4] = ${hcpv.module.mjeff4};
	arr[3] = ${hcpv.module.mjeff3};
	arr[2] = ${hcpv.module.mjeff2};
	arr[1] = ${hcpv.module.mjeff1};
	arr[0] = ${hcpv.module.mjeff0};
	return arr;
};

equations{ 'module_rad' } = define() {
	arr[4] = ${hcpv.module.rad4};
	arr[3] = ${hcpv.module.rad3};
	arr[2] = ${hcpv.module.rad2};
	arr[1] = ${hcpv.module.rad1};
	arr[0] = ${hcpv.module.rad0};
	return arr;
};

function cpv_replot_airmass()
{
	a0 = value('module_a0');
	a1 = value('module_a1');
	a2 = value('module_a2');
	a3 = value('module_a3');
	a4 = value('module_a4');
	np = 100;	
	TZ = alloc(np);
	MAM = alloc(np);
	
	for (i=0;i<np;i++)
	{
		TZ[i] = i*90.0/np;
		AM = 1/cosd(TZ[i]);
		MAM[i] = a0 + a1*AM + a2*pow(AM,2) + a3*pow(AM,3) + a4*pow(AM,4);
		//log("["+ i + "]: TZ,MAM=" + TZ[i] + " " + MAM[i] );
	}
	
	setplot("hcpv_module_mam_plot");
	clearplot('hcpv_module_mam_plot');
	//plot("hcpv_module_mam_plot", TZ, MAM, {"thick"=1, "scale"=true});
	plot( TZ, MAM, {"thick"=1, "scale"=true});
	axis( "x1", {"min"=0, "max"=90, "label"="Zenith angle"});
	axis( "y1", {"min"=0, "max"=1.3, "label"="MAM"});

	//plot( ws, weibull_probability, {"thick"=1, "scale"=true, "color"=[100,200,100], "series"="Weibull"});
	//axis( "x1", {"min"=0,  "label"="Wind Speed"});

	//newplot("hcpv.module.mam_plot");
}
