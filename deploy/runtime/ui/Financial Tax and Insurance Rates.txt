function enable_sales_tax() 
{

	str = 'The sales tax rate applies to the total direct cost on the System Costs page.';
	color = 'black';
	
	tech = technology();
	if (tech=='Flat Plate PV' || tech=='PVWatts') {
		capex_cost_choice = value('pv_capex_cost_choice');
		enable('sales_tax_rate', capex_cost_choice!=1);
		if ( capex_cost_choice == 1 ) {
			str = 'Sales tax is disabled for the capital cost curve option on the System Costs page. Be sure to include sales tax in the costs you specify on the System Costs page.';
			color = 'black';
		}	
	}

	property( 'capex_table_sales_tax_warning', 'Caption', str );
	property( 'capex_table_sales_tax_warning', 'TextColour', color );
}

on_load{'Financial Tax and Insurance Rates'} = define() {
	enable_sales_tax();
};
