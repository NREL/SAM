equations {'itemized_check_other_infras'} = define() {
	if (${balance_of_system_cost_override}!=1)
	{
		return ?${other_infrastructure_cost_method}
				[
					0,
					0,
					2,
					1
				];
	}
	else
	{
		return 2;
	}
};

on_load{'Marine Costs - Other Infrastructure'} = define() {
	on_change{'itemized_check_other_infras'};
	enable_itemized_other_infras();
	check_override_other_infras();
	on_change{'other_infras_override'};
};

on_change{'other_infras_override'} = define() {
	check_override_other_infras();
};

function check_override_other_infras()
{
	check_value = value('other_infras_override');
	//enable('other_infrastructure_cost_method',check_value!=1);
	//enable('other_infrastructure_cost',check_value!=1);
	show('other_infras_override_label',check_value==1);
	
	refresh();
}	

on_change{'itemized_check_other_infras'} = define() {
	enable_itemized_other_infras();
};

function enable_itemized_other_infras()
{
	itemized_value = value('itemized_check_other_infras');
	enable('offshore_accomodations_other_infras_cost',itemized_value==1);
	enable('dedicated_om_vessel_cost',itemized_value==1);
	enable('onshore_om_facilities_cost',itemized_value==1);
	enable('om_equipment_purchases_cost',itemized_value==1);
	enable('offshore_logistical_infrastructure_other_infras_cost',itemized_value==1);
	enable('other_infrastructure_transportation_other_infras_cost',itemized_value==1);
	enable('other_other_infras_cost',itemized_value==1);
	enable('other_infrastructure_cost_input',itemized_value!=2);
	show('other_infrastructure_cost_total',itemized_value==1);
	show('other_level_label',itemized_value==1);
	show('other_infrastructure_cost_input',(itemized_value==0 || itemized_value==2));
	show('other_user_input_label',(itemized_value==0 || itemized_value==2));
	//show('level_label',itemized_value==1);
	//show('level_line',itemized_value==1);
	
	/*if (value('itemized_check_eng')==1)
	{
		enable_itemized_level_eng();
	}*/
	refresh();
}
