equations{ 'F_wc' } = define() {
	retval=[
		${csp.dtr.pwrb.hc_ctl1},
		${csp.dtr.pwrb.hc_ctl2},
		${csp.dtr.pwrb.hc_ctl3},
		${csp.dtr.pwrb.hc_ctl4},
		${csp.dtr.pwrb.hc_ctl5},
		${csp.dtr.pwrb.hc_ctl6},
		${csp.dtr.pwrb.hc_ctl7},
		${csp.dtr.pwrb.hc_ctl8},
		${csp.dtr.pwrb.hc_ctl9}];
	return retval;
};

equations{ 'tech_type' } = define() {
	if (${csp.dtr.pwrb.pressure_mode} == 1) //sliding pressure
		return 3; // user defined
	else // fixed pressure
		return 2; // trough
};

equations{ 'CT' } = define() {
	return ${combo_condenser_type}+1;
};

equations{ 'fossil_mode' } = define() {
	return ${comb_fossil_mode}+1;
};

equations{ '$MIMO$ efficiency calculations' } = define() {
	gross = ${pb_rated_cap};
	${W_pb_design} = gross;

	qdesign = (${eta_ref}!=0) ? gross/${eta_ref} : gross ;
	${q_pb_design} = qdesign;
	${q_max_aux} = qdesign;
};

equations{ 'csp.dtr.pwrb.nameplate' } = define() {
	return ${pb_rated_cap}*${gross_net_conversion_factor}; };

equations{ 'csp.dtr.pwrb.design_inlet_temp' } = define() {
	return ${T_loop_out}; };

equations{ 'csp.dtr.pwrb.design_outlet_temp' } = define() {
	return ${T_loop_in_des};
};

equations{ 'system_capacity' } = define() {
	return ${csp.dtr.pwrb.nameplate} * 1000.0; // MW to kW
};

on_load{'Physical Trough Power Block'} = define() {
	
	on_change{'combo_condenser_type'}();
};

on_change{'combo_condenser_type'} = define() {
  state=value('combo_condenser_type');
  //evaporative or hybrid
  enable('dT_cw_ref', ( state == 0 || state == 2 ) ); 
  enable('T_approach', ( state == 0 || state == 2 ) );
  //air cooled or hybrid
  enable('T_ITD_des', ( state == 1 || state == 2 ) ); 
  enable('P_cond_ratio', ( state == 1 || state == 2) );
  //hybrid
  enable('csp.dtr.pwrb.hc_ctl1', state == 2);
  enable('csp.dtr.pwrb.hc_ctl2', state == 2);
  enable('csp.dtr.pwrb.hc_ctl3', state == 2);
  enable('csp.dtr.pwrb.hc_ctl4', state == 2);
  enable('csp.dtr.pwrb.hc_ctl5', state == 2);
  enable('csp.dtr.pwrb.hc_ctl6', state == 2);
  enable('csp.dtr.pwrb.hc_ctl7', state == 2);
  enable('csp.dtr.pwrb.hc_ctl8', state == 2);
  enable('csp.dtr.pwrb.hc_ctl9', state == 2);
  //default min condenser pressure
  if ( state == 0 ) value('P_cond_min', 1.25); //evaporative
  if ( state == 1 || state == 2 ) value('P_cond_min', 2); //air cooled or hybrid
};
