equations{ 'hcpv.array.single_tracker_nameplate' } = define() {
	return ${array_modules_per_tracker}*${hcpv.module.power}/1000; };

equations{ 'hcpv.array.total_land_area' } = define() {
	return ${hcpv.module.area}*${array_modules_per_tracker}*${array_num_trackers} * ${hcpv.array.packing_factor} * 0.0002471; };

equations{ 'hcpv.array.overall_est_eff' } = define() {
	return ${hcpv.module.est_eff}
 * ${array_tracking_error}
 * ${array_dc_wiring_loss}
 * ${hcpv.array.average_soiling}
 * ${array_dc_mismatch_loss}
 * ${array_diode_conn_loss}
 * ${array_ac_wiring_loss}
 * ${inv_snl_paco}/${inv_snl_pdco}; };

equations{ 'hcpv.array.tracker_power' } = define() {
	return ${array_tracker_power_fraction}*${hcpv.array.single_tracker_nameplate}*1000; };

equations{ 'hcpv.array.nameplate' } = define() {
	return ${array_modules_per_tracker}*${array_num_trackers}*${hcpv.module.power}/1000; };

equations{ 'hcpv.array.average_soiling' } = define() {
	return 0.95; };

equations{ 'array_num_inverters' } = define() {
	return ceil( ${hcpv.array.nameplate} / (${inv_snl_pdco}/1000) ); };

equations{ 'hcpv.array.ac_capacity' } = define() {
	return ${array_num_inverters}*${inv_snl_paco}/1000; };

equations{ 'system_capacity' } = define() {
	return ${hcpv.array.nameplate}; };

function loss_adjustment_set() 
{
	if (financing() != "LCOE Calculator" && financing() != "None") {
		if (value('system_use_lifetime_output') == 1) {
			property('adjust', 'AnalysisPeriod', value('analysis_period'));
		}
		else {
			property('adjust', 'AnalysisPeriod', 1);
		}
	}
	else {
		property('adjust', 'AnalysisPeriod', 1);
	}
}


on_load{'HCPV Array'} = define() 
{
	loss_adjustment_set();
};
