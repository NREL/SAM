on_load{'Marine Availability'} = define() {
	enable_avail();
};

function enable_avail()
{
	if (financing() != "Single Owner") {
		check = value('wave_resource_model_choice');
		if (check == 1) {
			enable('me_adjust', true);
			property('me_avail_label', 'Caption', 'System availability losses reduce the system output to represent system outages or other events. Losses are applied in addition to the constant loss factors specified above');
		}
		else {
			enable('me_adjust', false);
			property('me_avail_label', 'Caption', 'System availability losses are only available for time series wave resource data.');
			
		}
	}
	else {
		enable('me_adjust', true);
		property('me_avail_label', 'Caption', 'System availability losses reduce the system output to represent system outages or other events. Losses are applied in addition to the constant loss factors specified above');
	}
}
