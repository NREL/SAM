on_load{'Operating Costs'} = define() {
	enable( 'om_capacity_escal',#${om_capacity}==1);
	enable( 'om_fixed_escal',#${om_fixed}==1);
	enable( 'om_production_escal',#${om_production}==1);
	property( 'om_capacity','FixedLength',${analysis_period});
	property( 'om_fixed','FixedLength',${analysis_period});
	property( 'om_production','FixedLength',${analysis_period});
};

on_change{'om_capacity'} = define() {
	enable( 'om_capacity_escal',#${om_capacity}==1);
};

on_change{'om_fixed'} = define() {
	enable( 'om_fixed_escal',#${om_fixed}==1);
};

on_change{'om_production'} = define() {
	enable( 'om_production_escal',#${om_production}==1);
};
