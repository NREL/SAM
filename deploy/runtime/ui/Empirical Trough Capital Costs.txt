// direct cost equations
equations{ 'csp.tr.cost.site_improvements.area' } = define() {
	return ${Solar_Field_Area}; };
equations{ 'csp.tr.cost.site_improvements' } = define() {
	return ${Solar_Field_Area}*${csp.tr.cost.site_improvements.cost_per_m2}; };

equations{ 'csp.tr.cost.solar_field.area' } = define() {
	return ${Solar_Field_Area}; };
equations{ 'csp.tr.cost.solar_field' } = define() {
	return ${Solar_Field_Area}*${csp.tr.cost.solar_field.cost_per_m2}; };

equations{ 'csp.tr.cost.htf_system.area' } = define() {
	return ${Solar_Field_Area}; };
equations{ 'csp.tr.cost.htf_system' } = define() {
	return ${Solar_Field_Area}*${csp.tr.cost.htf_system.cost_per_m2}; };

equations{ 'csp.tr.cost.storage.mwht' } = define() {
	return ${calc_max_energy}; };
equations{ 'csp.tr.cost.storage' } = define() {
	return ${calc_max_energy}*${csp.tr.cost.storage.cost_per_kwht}*1000; };

equations{ 'csp.tr.cost.power_plant.mwe' } = define () {
    return ${TurbOutG}; };
equations{ 'csp.tr.cost.power_plant' } = define() {
	return ${TurbOutG}*${csp.tr.cost.power_plant.cost_per_kwe}*1000; };

equations{ 'csp.tr.cost.fossil_backup.mwe' } = define () {
    return ${TurbOutG}; };
equations{ 'csp.tr.cost.fossil_backup' } = define() {
	return ${TurbOutG}*${csp.tr.cost.fossil_backup.cost_per_kwe}*1000; };

equations{ 'csp.tr.cost.bop.mwe' } = define () {
    return ${TurbOutG}; };
equations{ 'csp.tr.cost.bop' } = define() {
	return ${TurbOutG}*${csp.tr.cost.bop_per_kwe}*1000; };

equations{ 'csp.tr.cost.contingency' } = define() {
	return ${csp.tr.cost.contingency_percent}/100*(
	 ${csp.tr.cost.site_improvements}
	+${csp.tr.cost.solar_field}
	+${csp.tr.cost.htf_system}
	+${csp.tr.cost.storage}
	+${csp.tr.cost.fossil_backup}
	+${csp.tr.cost.power_plant}
	+${csp.tr.cost.bop} ); };



equations{ 'total_direct_cost' } = define() {
	return ${csp.tr.cost.contingency}
+${csp.tr.cost.site_improvements}
+${csp.tr.cost.solar_field}
+${csp.tr.cost.htf_system}
+${csp.tr.cost.storage}
+${csp.tr.cost.fossil_backup}
+${csp.tr.cost.power_plant}
+${csp.tr.cost.bop}; };

//indirect cost equations
equations{ 'csp.tr.cost.total_land_area' } = define() {
	return ${ui_total_land_area}; };
equations{ 'csp.tr.cost.nameplate' } = define() {
	return ${ui_net_capacity}; };	

equations{ 'csp.tr.cost.epc.total' } = define() {
	return ${csp.tr.cost.epc.per_acre}*${csp.tr.cost.total_land_area} 
+ ${csp.tr.cost.epc.percent}/100*${total_direct_cost} 
+ ${system_capacity}*1000000*${csp.tr.cost.epc.per_watt} 
+ ${csp.tr.cost.epc.fixed}; };

equations{ 'csp.tr.cost.plm.total' } = define() {
	return ${csp.tr.cost.plm.per_acre}*${csp.tr.cost.total_land_area} 
+ ${csp.tr.cost.plm.percent}/100*${total_direct_cost} 
+ ${system_capacity}*1000000*${csp.tr.cost.plm.per_watt} 
+ ${csp.tr.cost.plm.fixed}; };

equations{ 'csp.tr.cost.sales_tax.value' } = define() {
	return ${sales_tax_rate}; };
equations{ 'csp.tr.cost.sales_tax.total' } = define() {
    return ${sales_tax_rate}/100*${total_direct_cost}*${csp.tr.cost.sales_tax.percent}/100; };

equations{ 'total_indirect_cost' } = define() {
	return ${csp.tr.cost.epc.total}+${csp.tr.cost.plm.total}; };

//total cost equations
equations{ 'total_installed_cost' } = define() {
	return ${total_direct_cost}+${total_indirect_cost}+${csp.tr.cost.sales_tax.total}; };
	
equations{ 'csp.tr.cost.installed_per_capacity' } = define() {
	return 0.001*${total_installed_cost}/${ui_net_capacity}; };

	return 0;};
