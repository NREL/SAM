	return ${nMod} * ${nLoops} * ${SCA_drives_elec}; };

	return ${csp.mslf.control.bop_array_mult}*${csp.mslf.control.bop_array_pf}
	*(${csp.mslf.control.bop_array_c0}+${csp.mslf.control.bop_array_c1}
	+${csp.mslf.control.bop_array_c2})*${P_ref}
; };

	return ${csp.mslf.control.aux_array_mult}*${csp.mslf.control.aux_array_pf}

; };
