on_load{'Marine Costs - O&M'} = define() {
	
	//Operations
	show('operations_cost_total',value('operations_cost_method')==3);
	show('operations_cost_input',value('operations_cost_method')!=3);
	enable('operations_cost_input',value('operations_cost_method')!=2);
	
	//Maintenance
	show('maintenance_cost_total',value('maintenance_cost_method')==3);
	show('maintenance_cost_input',value('maintenance_cost_method')!=3);
	enable('maintenance_cost_input',value('maintenance_cost_method')!=2);
	
	on_change{'om_cost_override'};
	override_om_cost();
	
	refresh();
	
};

on_change{'om_cost_override'} = define() {
	override_om_cost();
	
};

function override_om_cost() {
	override_value = value('om_cost_override');
	enable('total_operating_cost',override_value!=1);
	enable('om_cost_input',override_value==1);
	show('om_cost_input',override_value==1);
	show('total_operating_cost',override_value!=1);
	
	//Operations
	enable('operations_cost_method',override_value==0);
	if (value('operations_cost_method') < 2) {
		enable('operations_cost_input', override_value==0);
	}
	//show('operations_override_label',override_value==1);
	
	//Maintenance
	enable('maintenance_cost_method',override_value==0);
	if (value('maintenance_cost_method') < 2) {
		enable('maintenance_cost_input', override_value==0);
	}
	//show('maintenance_override_label',override_value==1);
	
	
	refresh();
}
