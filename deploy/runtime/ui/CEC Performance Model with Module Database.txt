function cec_module_update_ui()
{
	state = value('cec_temp_corr_mode');
	ht = value('cec_heat_transfer');	
	mc = value('cec_mounting_config');
		
	enable( 'cec_standoff', state == 0 );
	enable( 'cec_height', state == 0 );
	
	enable( 'cec_mounting_config', state == 1 );
	enable( 'cec_heat_transfer', state == 1 );
	enable( 'cec_mounting_orientation', state == 1 && mc == 3 );
	enable( 'cec_module_width', state == 1 );
	enable( 'cec_array_rows', state == 1 && ht == 1 );
	enable( 'cec_array_cols', state == 1 && ht == 1 );
	enable( 'cec_backside_temp', state == 1 && mc == 2 );
	enable( 'cec_gap_spacing', state == 1 && mc == 3);
	
	enable('cec_bifaciality', value('cec_is_bifacial'));
	enable('cec_bifacial_transmission_factor', value('cec_is_bifacial'));
	enable('cec_bifacial_ground_clearance_height', value('cec_is_bifacial'));
	
	show('lbl_bifacial', value('lib_is_bifacial') && !value('cec_is_bifacial'));
	refresh();

}

function cec_module_plot()
{
	//assigning module specifications
	i_l_ref = ${cec_i_l_ref};
	i_o_ref = ${cec_i_o_ref};
	r_s = ${cec_r_s};
	a_ref = ${cec_a_ref};
	r_sh_ref = ${cec_r_sh_ref};
	i_mp_ref = ${cec_i_mp_ref};
	//allocations and size of array
	array_size = 602;
	interval = ${cec_v_mp_ref}/300;
	x= alloc(array_size); y= alloc(array_size);
	//First Point- Short Circuit
	x[0]= 0; y[0]= ${cec_i_sc_ref};
	i=1;
	//Before Max Power Point (MPP)
	while(i<300)
	{
		Vcec= i*interval;
		Icec= current_at_voltage_cec(Vcec, i_l_ref, i_o_ref, r_s, a_ref, r_sh_ref, i_mp_ref);
		x[i] = Vcec; y[i] = Icec;
		i= i + 1;
	}
	//MPP
	x[i] = ${cec_v_mp_ref}; y[i] = i_mp_ref;
	i= i + 1;
	//After MPP
	interval = (${cec_v_oc_ref} - ${cec_v_mp_ref})/300;
	while(i<601)
	{
		Vcec = ${cec_v_mp_ref} + (i-300) * interval;
		Icec = current_at_voltage_cec(Vcec, i_l_ref, i_o_ref, r_s, a_ref, r_sh_ref, i_mp_ref );
		x[i] = Vcec; y[i] = Icec;
		i= i + 1;
	}
	//Last Point- Open Circuit
	x[i] = ${cec_v_oc_ref}; y[i] = 0;
	//Make the plot
	setplot("pv.mod.cec.iv_curve");
	clearplot("pv.mod.cec.iv_curve");
	plotopt({"title"=${module_name},"popup"=true,"backcolor"=[255,255,255],"legend"=false});
	plot(x,y,{"thick"=2, "scale"=true,"color"=[200,100,100]});
	axis('x1', {'label'='Module Voltage (Volts)', 'min'=0, 'max'=ceil(${cec_v_oc_ref})});
	axis('y1', {'label'='Module Current (Amps)', 'min'=0, 'max'= (max(y) + 0.5)});
}

on_load{"CEC Performance Model with Module Database"} = define()
{
	cec_module_update_ui();
	cec_module_plot();
};
	
on_change{'cec_temp_corr_mode'} = cec_module_update_ui;

on_change{'cec_heat_transfer'} = cec_module_update_ui;
on_change{'cec_is_bifacial'} = cec_module_update_ui;
on_change{'module_name'} = define()
{
	cec_module_update_ui();
	cec_module_plot();
};
