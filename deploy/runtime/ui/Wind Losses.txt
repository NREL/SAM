	if (${wind_farm_wake_model} == 3)
		return ${wake_loss};

		return 0;

	return (1-((1-${wake_int_loss}/100) * (1-${wake_ext_loss}/100) * (1-${wake_future_loss}/100)))*100;

	return (1-((1-${avail_bop_loss}/100) * (1-${avail_grid_loss}/100) * (1-${avail_turb_loss}/100)))*100;
