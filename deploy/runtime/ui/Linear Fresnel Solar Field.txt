equations {'total_land_area'} = define() {
	return ${csp.lf.sf.total_land_area};
};

equations{ 'P_turb_des' } = define() {
	return ${P_boil_des}; };

equations{ 'csp.lf.sf.dp.actual_aper' } = define() {
	return ${csp.lf.sf.dp.loop_aperture} * ${nLoops}; };

equations{ 'csp.lf.sf.total_land_area' } = define() {
	return ${csp.lf.sf.field_area}*${csp.lf.sf.area_multiplier}; };

equations{ 'csp.lf.sf.dp.loop_aperture' } = define() {
	return ?${csp.lf.sf.sh_geom_unique} [
/* single geometry (sh_geom_unique = 0) */
	(${nModBoil}+${nModSH}) * ${csp.lf.geom1.refl_aper_area} , 
/* multiple geometry (sh_geom_unique = 1) */
	  ${nModBoil}*${csp.lf.geom1.refl_aper_area} 
	+ ${nModSH}*${csp.lf.geom2.refl_aper_area}
]; };

equations{ 'q_max_aux' } = define() {
	return  1.e-6 * ${csp.lf.sf.dp.actual_aper} 
* ${I_bn_des}
* ${csp.lf.sf.dp.total_loop_conv_eff}; };

equations{ 'csp.lf.sf.geom2_area_frac' } = define() {
	return ?${csp.lf.sf.sh_geom_unique} [
	0,
	${nModSH}*${csp.lf.geom2.refl_aper_area}
	/ (${nModBoil}*${csp.lf.geom1.refl_aper_area} + ${nModSH}*${csp.lf.geom2.refl_aper_area} )
]; };

equations{ 'csp.lf.sf.geom1_area_frac' } = define() {
	return ?${csp.lf.sf.sh_geom_unique} [
	1,
	${nModBoil}*${csp.lf.geom1.refl_aper_area}
	/ (${nModBoil}*${csp.lf.geom1.refl_aper_area} + ${nModSH}*${csp.lf.geom2.refl_aper_area} )
]; };

equations{ 'csp.lf.sf.dp.sm1_aperture' } = define() {
	return ${demand_var}/${eta_ref}
/( ${I_bn_des}*${csp.lf.sf.dp.total_loop_conv_eff} )*1e6; };

equations{ 'csp.lf.sf.field_area' } = define() {
	return ${csp.lf.sf.dp.actual_aper}/4046.85642; };

equations{ 'csp.lf.sf.dp.sm1_numloops' } = define() {
	return ceil(${csp.lf.sf.dp.sm1_aperture}/${csp.lf.sf.dp.loop_aperture}); };

equations{ 'csp.lf.sf.total_pres_drop' } = define() {
	return ( ${fP_hdr_c}
 + ${fP_sf_boil}
 + ${fP_boil_to_sh}
 + ${fP_sf_sh}
 + ${fP_hdr_h}
 ) * ${P_turb_des}; };

equations{ 'csp.lf.sf.dp.loop_therm_eff' } = define() {
	return ${csp.lf.geom1.rec_thermal_derate}*${csp.lf.sf.geom1_area_frac} + ${csp.lf.geom2.rec_thermal_derate}*${csp.lf.sf.geom2_area_frac}; };

equations{ 'latitude' } = define() {
	return ${lat};
};

equations{ 'azimuth' } = define() {
	return ${ColAz};
};

equations{ 'csp.lf.sf.dp.piping_therm_eff' } = define() {
	return 1-${Pipe_hl_coef}*( (${T_cold_ref}+${T_hot})/2-${T_amb_des_sf})/${I_bn_des}; };

equations{ 'csp.lf.sf.dp.total_loop_conv_eff' } = define() {
	return ${csp.lf.sf.dp.loop_opt_eff}*${csp.lf.sf.dp.loop_therm_eff}*${csp.lf.sf.dp.piping_therm_eff}; };

equations{ 'csp.lf.sf.dp.loop_opt_eff' } = define() {
	return  ${csp.lf.sf.geom1_area_frac} * ${csp.lf.geom1.rec_optical_derate}*${csp.lf.geom1.coll_opt_loss_norm_inc}
  + ${csp.lf.sf.geom2_area_frac} * ${csp.lf.geom2.rec_optical_derate}*${csp.lf.geom2.coll_opt_loss_norm_inc} ; };

equations{ 'solarm' } = define() {
	return ?${csp.lf.sf.sm_or_area}
[
	${csp.lf.sf.specified_solar_multiple},
	${csp.lf.sf.dp.actual_aper}/${csp.lf.sf.dp.sm1_aperture}
]; };

equations{ 'nLoops' } = define() {
	return to_int(
?${csp.lf.sf.sm_or_area}
[
	${csp.lf.sf.specified_solar_multiple}*${csp.lf.sf.dp.sm1_aperture},
	${csp.lf.sf.specified_total_aperture}
]
/ ${csp.lf.sf.dp.loop_aperture}  ); };

function update_dslf_sf_ui(){
	mode = value("csp.lf.geom2.opt_mode");
	mode2 = value("csp.lf.geom1.opt_mode");
	enable("ColAz",(mode!=0)||(mode2!=0));
}

	on_change{"csp.lf.sf.sm_or_area"}();
	on_change{'is_oncethru'}();
	update_dslf_sf_ui();
		};

	enable('x_b_des',mode==0);
	};
