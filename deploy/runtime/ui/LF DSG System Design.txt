equations{ 'specified_q_dot_rec_des' } = define() 
{
	return ${specified_solar_multiple}*${q_pb_des};

{
	return ${specified_q_dot_rec_des}*1000.0;
};

equations{ '$MIMO$ ui_lf_dsg_design' } = define()
{
	obj = ssc_create();

	ssc_var(obj, "P_boil", ${P_turb_des});
	ssc_var(obj, "use_quality_or_subcooled", ${use_quality_or_subcooled});
	ssc_var(obj, "deltaT_subcooled", ${deltaT_subcooled});

	ret = ssc_exec(obj, 'csp_dsg_lf_ui');

	//ret_d = ssc_dump(obj, 'etes_system_des_mimo_ssc_auto_exec.txt');

	if(ret==0) {

		${T_hot} = ssc_var(obj, "T_saturation");	//[C]
		${T_hot_out_target} = ssc_var(obj, "T_hot_out_target");
	}
	else {
		vf = -789;
		${T_hot} = vf;
		${T_hot_out_target} = vf;
	}
	
	ssc_free( obj );
};

	on_change{'use_quality_or_subcooled'}();

on_change{'use_quality_or_subcooled'} = define() 
{
  state = value('use_quality_or_subcooled');
  enable ('x_b_des', state == 0);
  enable ('deltaT_subcooled', state == 1);
};
