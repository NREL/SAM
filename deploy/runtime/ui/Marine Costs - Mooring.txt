	if (${device_cost_override}!=1)

		return ?${mooring_found_substruc_cost_method}

					2,

	{
		return 2;
	}

			];
};

on_load{'Marine Costs - Mooring'} = define() {
	check_override_mooring();
	on_change{'mooring_override'};
	on_change{'itemized_check_mooring'};
	enable_itemized_mooring();
	on_change{'detailed_cost_choice_outfitting_structure'};
	on_change{'detailed_cost_choice_ancillary_marine_systems'};
	detailed_cost_outfitting_structure();
	detailed_cost_ancillary_marine_systems();
};

on_change{'mooring_override'} = define() {
	check_override_mooring();
};

function check_override_mooring()
{
	check_value = value('mooring_override');
	//enable('mooring_found_substruc_cost_input',check_value!=1);
	//enable('mooring_found_substruc_cost_total',check_value!=1);
	//enable('mooring_found_substruc_cost_method',check_value!=1);
	//enable('mooring_found_substruc_cost_modeled',check_value!=1);
	//enable('mooring_found_substruc_cost',check_value!=1);
	show('mooring_override_label',check_value==1);
	refresh();
}

on_change{'detailed_cost_choice_outfitting_structure'} = define() {
	detailed_cost_outfitting_structure();
};

function detailed_cost_outfitting_structure()
{
	if (value('itemized_check_mooring')==1)
	{
		detailed_value = value('detailed_cost_choice_outfitting_structure');
		enable('outfitting_structure_cost',detailed_value==0);
		show('outfitting_structure_cost',detailed_value==0);
		enable('outfitting_structure_cost_total',detailed_value);
		show('outfitting_structure_cost_total',detailed_value);
		enable('vessel_landing_cost',detailed_value);
		enable('service_platforms_and_decks_cost',detailed_value);
		enable('ladders_cost',detailed_value);
		enable('railings_cost',detailed_value);
		enable('marine_coatings_cost',detailed_value);
		enable('other_outfitting_structure_cost',detailed_value);
		
		refresh();
	}
}

on_change{'detailed_cost_choice_ancillary_marine_systems'} = define() {
	detailed_cost_ancillary_marine_systems();
};

function detailed_cost_ancillary_marine_systems()

		enable('ancillary_marine_systems_cost',detailed_value==0);
		show('ancillary_marine_systems_cost',detailed_value==0);
		enable('ancillary_marine_systems_cost_total',detailed_value);
		show('ancillary_marine_systems_cost_total',detailed_value);

		enable('condition_monitoring_cost',detailed_value);
		enable('other_ancillary_marine_system_cost',detailed_value);
		
		refresh();
	}
}
