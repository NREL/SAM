equations{ 'csp.pt.cost.receiver.area' } = define() {
	tech = technology();
	area = 0;
	if (tech == 'MSPT' || tech == 'ISCC')
	{
		area = ?${receiver_type}
		[
		${H_rec}*${D_rec}*3.1415926, /*external*/
		${rec_d_spec}*${csp.pt.rec.cav_ap_height}
		];
	}
	else if (tech == 'DSPT')
	{
		area =  ${d_rec}*${H_rec}*3.1415926;
	}
	return area; };
