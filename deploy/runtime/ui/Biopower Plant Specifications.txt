on_load{"Biopower Plant Specifications"}=define(){

	on_change{"biopwr.plant.tou_option"}();
	loss_adjustment_set();
};
	
on_change{"biopwr.plant.drying_method"} = define()
{
	dryopt=value("biopwr.plant.drying_method");
	enable("biopwr.plant.drying_spec_wet", dryopt==2);
	enable("biopwr.cost.dryer_per_kw", dryopt==2);
	enable("biopwr.cost.dryer_capacity", dryopt==2);
	enable("biopwr.cost.dryer", dryopt==2);
};


on_change{"biopwr.plant.ramp_opt"} = define()
{
	ramp=value("biopwr.plant.ramp_opt");
	enable("biopwr.plant.ramp_opt1", ramp==1);
	enable("biopwr.plant.ramp_opt2", ramp==2);
};

on_change{"biopwr.plant.tou_option"} = define()
{
	tou=value("biopwr.plant.tou_option");
	ramp=value("biopwr.plant.ramp_opt");
	enable("biopwr.plant.tou_grid", tou==1);
	enable("biopwr.plant.ramp_opt", tou==1);
	enable("biopwr.plant.ramp_opt1", tou==1 && ramp==1);
	enable("biopwr.plant.ramp_opt2", tou==1 && ramp==2);
	for ( i = 1; i<10; i++) {
		enable("biopwr.plant.disp"+i+".power", tou==1);
	}

};

function loss_adjustment_set() 
{
	if (financing() != "LCOE Calculator" && financing() != "None") {
		if (value('system_use_lifetime_output') == 1) {
			property('adjust', 'AnalysisPeriod', value('analysis_period'));
		}
		else {
			property('adjust', 'AnalysisPeriod', 1);
		}
	}
	else {
		property('adjust', 'AnalysisPeriod', 1);
	}
}
