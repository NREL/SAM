	set_chemistry();

	show('batt_bank_duration',!is_cap);
	show('batt_bank_size',is_cap);
	refresh();

	on_change{'bank_bank_size_specify'};

	on_change{'batt_bank_duration'};
};

function set_bank_sizing()
{
	isauto = (value('batt_size_choice') == 0);
	isflow = is_flow();

	// auto-sizing
	enable('batt_bank_size', isauto);
	enable('batt_bank_power', isauto);
	enable('batt_bank_voltage', isauto);
	enable('batt_bank_size_dc_ac', isauto);
	enable('batt_duration_choice', isauto);
	enable('batt_bank_duration', isauto);
	// manual specify
	show('batt_bank_size_specify', isflow);
	show('batt_C_rate_max_charge_input', !isflow);
	show('batt_C_rate_max_discharge_input', !isflow);

	enable('batt_bank_ncells_serial', !isauto);
	enable('batt_bank_nstrings', !isauto);
	enable('batt_bank_nseries_stacks', !isauto);
	

	if (isflow)
		enable('batt_bank_size_specify', !isauto);
	else
	{
		enable('batt_C_rate_max_charge_input', !isauto);
		enable('batt_C_rate_max_discharge_input', !isauto);
	}
}

function set_duration_choice()
{
	is_cap = (value('batt_duration_choice') == 0);
	
	show('batt_bank_duration',!is_cap);
	show('batt_bank_size',is_cap);
	bank_power = value('batt_bank_power');
	bank_cap = value('batt_bank_size');
	if (!is_cap)
	{
		duration = bank_cap / bank_power;
		value('batt_bank_duration', duration);
		bank_cap_calc = duration * bank_power;
		value('batt_bank_size',bank_cap_calc);
	}
	else
	{
		duration = value('batt_bank_duration');
		bank_cap_calc = duration * bank_power;
		value('batt_bank_size', bank_cap_calc);
	}
	
	refresh();
}

function calc_batt_size() {
	is_cap = (value('batt_duration_choice') == 0);
	if (is_cap) {
		bank_cap = value('batt_bank_size');
		bank_power = value('batt_bank_power');
		bank_duration = bank_cap / bank_power;
		value('batt_bank_duration',batt_duration);
	}
	else {
		bank_duration = value('batt_bank_duration');
		bank_power = value('batt_bank_power');
		bank_cap = bank_power * bank_duration;
		value('batt_bank_size', bank_cap);
	}
}
