equations{ 'csp.ds.cost.engine' } = define() {
	return ${csp.ds.ncollectors}*${csp.ds.cost.engine.kw}*${csp.ds.cost.engine.cost_per_kw}; };

equations{ 'csp.ds.cost.sales_tax.total' } = define() {
	return ${csp.ds.cost.sales_tax.value}/100*${total_direct_cost}*${csp.ds.cost.sales_tax.percent}/100; };

equations{ 'csp.ds.cost.collector.area' } = define() {
	return ${A_proj}; };

equations{ 'csp.ds.cost.sales_tax.value' } = define() {
	return ${sales_tax_rate}; };

equations{ 'csp.ds.cost.plm.total' } = define() {
	return ${csp.ds.cost.plm.per_acre}*${csp.ds.cost.total_land_area} 
+ ${csp.ds.cost.plm.percent}/100*${total_direct_cost} 
+ ${csp.ds.cost.nameplate}*1000*${csp.ds.cost.plm.per_watt} 
+ ${csp.ds.cost.plm.fixed}; };

equations{ 'csp.ds.cost.receiver.kw' } = define() {
	return ${csp.ds.nameplate_capacity}; };

equations{ 'csp.ds.cost.engine.kw' } = define() {
	return ${csp.ds.nameplate_capacity}; };

equations{ 'total_direct_cost' } = define() {
	return ${csp.ds.cost.site_improvements}
+${csp.ds.cost.collector}
+${csp.ds.cost.receiver}
+${csp.ds.cost.engine}
+${csp.ds.cost.contingency}; };

equations{ 'total_indirect_cost' } = define() {
	return ${csp.ds.cost.epc.total}+${csp.ds.cost.plm.total}; };

equations{ 'total_installed_cost' } = define() {
	return ${total_direct_cost}+${total_indirect_cost}+${csp.ds.cost.sales_tax.total}; };

equations{ 'csp.ds.cost.receiver' } = define() {
	return ${csp.ds.ncollectors}*${csp.ds.cost.receiver.kw}*${csp.ds.cost.receiver.cost_per_kw}; };

equations{ 'csp.ds.cost.site_improvements' } = define() {
	return ${csp.ds.cost.site_improvements.area}*${csp.ds.cost.site_improvements.cost_per_m2}; };

equations{ 'csp.ds.cost.collector' } = define() {
	return ${csp.ds.ncollectors}*${csp.ds.cost.collector.area}*${csp.ds.cost.collector.cost_per_m2}; };

equations{ 'csp.ds.cost.contingency' } = define() {
	return ${csp.ds.cost.contingency_percent}/100*(
	 ${csp.ds.cost.site_improvements}
	+${csp.ds.cost.collector}
	+${csp.ds.cost.receiver}
	+${csp.ds.cost.engine}); };

equations{ 'csp.ds.cost.epc.total' } = define() {
	return ${csp.ds.cost.epc.per_acre}*${csp.ds.cost.total_land_area} 
+ ${csp.ds.cost.epc.percent}/100*${total_direct_cost} 
+ ${csp.ds.cost.nameplate}*1000*${csp.ds.cost.epc.per_watt} 
+ ${csp.ds.cost.epc.fixed}; };

equations{ 'csp.ds.cost.installed_per_capacity' } = define() {
	return ${total_installed_cost}/${csp.ds.cost.nameplate}; };

equations{ 'csp.ds.cost.total_land_area' } = define() {
	return ${csp.ds.field_area}*0.00024710538147 /* sq meters to acres */ ; };

equations{ 'csp.ds.cost.site_improvements.area' } = define() {
	return ${csp.ds.field_area}; };

equations{ 'csp.ds.cost.nameplate' } = define() {
	return ${csp.ds.total_capacity};
};

equations { 'system_use_recapitalization' } = define () {
	return 0;};
