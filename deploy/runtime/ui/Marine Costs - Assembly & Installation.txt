equations {'substructure_assem_install_cost_total'} = define() {
	return ${foundation_assem_install_cost}+
		${substructure_assem_install_cost}+${scour_protection_assem_install_cost}+
		${noise_mitigation_cost}+${other_substructure_cost};
};

equations {'substructure_assem_install_cost_total_check'} = define() {
	if (${detailed_cost_choice_substructures}==0)
	{
		return ${substructures_and_foundations_cost};
	}
	else
	{
		return ${substructure_assem_install_cost_total};
	}
};



equations {'marine_energy_converter_device_cost_total'} = define() {
	return ${structural_assembly_assem_install_cost}+${power_takeoff_system_assem_install_cost}+
		${other_marine_energy_converter_device_cost};
};

equations {'marine_energy_converter_device_cost_total_check'} = define() {
	if (${detailed_cost_choice_mec_device}==0)
	{
		return ${marine_energy_converter_device_cost};
	}
	else
	{
		return ${marine_energy_converter_device_cost_total};
	}
};

equations {'offshore_electrical_infrastructure_cost_total'} = define() {
	return ${array_cables_assem_install_cost}+${offshore_export_cables_assem_install_cost}+
		${offshore_substation_assem_install_cost}+${offshore_accommodations_platform_cost}+
		${onshore_electric_infrastructure_assem_install_cost}+${other_offshore_electrical_infrastructure_cost};
};

equations {'offshore_electrical_infrastructure_cost_total_check'} = define() {
	if (${detailed_cost_choice_offshore_infrastructure}==0)
	{
		return ${offshore_electrical_infrastructure_cost};
	}
	else
	{
		return ${offshore_electrical_infrastructure_cost_total};
	}
};


equations {'assembly_and_install_cost_total'} = define() {
	return ${substructure_assem_install_cost_total_check}+${marine_energy_converter_device_cost_total_check}+
		${offshore_electrical_infrastructure_cost_total_check}+${offshore_logistical_infrastructure_assem_install_cost}+
		${other_assem_install_cost};
};

equations {'itemized_check_assem_install'} = define() {
	if (${balance_of_system_cost_override}!=1)
	{
		return ?${assembly_and_install_cost_method}
				[
					0,
					0,
					2,
					1
				];
	}
	else
	{
		return 2;
	}
};


equations {'assembly_and_install_cost'} = define() {
	return ?${assembly_and_install_cost_method}
			[
				${assembly_and_install_cost_input}*${system_capacity},
				${assembly_and_install_cost_input},
				${assembly_and_install_cost_modeled},
				${assembly_and_install_cost_total}
			];
};

	return ${balance_of_system_cost_override};};

on_load{'Marine Costs - Assembly & Installation'} = define() {
	on_change{'itemized_check_assem_install'};
	enable_itemized_assem_install();
	//enable_itemized_level();
	on_change{'detailed_cost_choice_substructures'};
	on_change{'detailed_cost_choice_mec_device'};
	on_change{'detailed_cost_choice_offshore_infrastructure'};
	detailed_cost_substructures();
	detailed_cost_mec_device();
	detailed_cost_offshore_infrastructure();
	check_override_assembly();
	on_change{'assembly_and_install_override'};
};

on_change{'assembly_and_install_override'} = define() {
	check_override_assembly();
};

function check_override_assembly()
{
	check_value = value('assembly_and_install_override');
	//enable('assembly_and_install_cost_method',check_value!=1);
	//enable('assembly_and_install_cost',check_value!=1);	
	show('assembly_override_label',check_value==1);
	
	refresh();
}

on_change{'detailed_cost_choice_substructures'} = define() {
	detailed_cost_substructures();
};

function detailed_cost_substructures()
{
	if (value('itemized_check_assem_install')==1)
	{
		detailed_value = value('detailed_cost_choice_substructures');
		enable('substructures_and_foundations_cost',detailed_value==0);
		show('substructures_and_foundations_cost',detailed_value==0);
		enable('substructure_assem_install_cost_total',detailed_value);
		show('substructure_assem_install_cost_total',detailed_value);
		enable('foundation_assem_install_cost',detailed_value);
		enable('substructure_assem_install_cost',detailed_value);
		enable('scour_protection_assem_install_cost',detailed_value);
		enable('noise_mitigation_cost',detailed_value);
		enable('other_substructure_cost',detailed_value);
		
		refresh();
	}
}

function enable_itemized_assem_install()
{
	itemized_value = value('itemized_check_assem_install');
	enable('substructures_and_foundations_cost',itemized_value==1);
	enable('substructure_assem_install_cost_total',itemized_value==1);

	enable('scour_protection_assem_install_cost',itemized_value==1);
	enable('noise_mitigation_cost',itemized_value==1);
	enable('other_substructure_cost',itemized_value==1);
	enable('marine_energy_converter_device_cost',itemized_value==1);
	enable('marine_energy_converter_device_cost_total',itemized_value==1);
	enable('detailed_cost_choice_mec_device',itemized_value==1);
	enable('structural_assembly_assem_install_cost',itemized_value==1);
	enable('power_takeoff_system_assem_install_cost',itemized_value==1);
	enable('other_marine_energy_converter_device_cost',itemized_value==1);
	enable('offshore_electrical_infrastructure_cost',itemized_value==1);
	enable('offshore_electrical_infrastructure_cost_total',itemized_value==1);
	enable('detailed_cost_choice_offshore_infrastructure',itemized_value==1);
	enable('array_cables_assem_install_cost',itemized_value==1);
	enable('offshore_export_cables_assem_install_cost',itemized_value==1);
	enable('offshore_substation_assem_install_cost',itemized_value==1);
	enable('offshore_accommodations_platform_cost',itemized_value==1);
	enable('onshore_electric_infrastructure_assem_install_cost',itemized_value==1);
	enable('other_offshore_electrical_infrastructure_cost',itemized_value==1);
	enable('offshore_logistical_infrastructure_assem_install_cost',itemized_value==1);
	enable('other_assem_install_cost',itemized_value==1);
	//show('level_eng',itemized_value==1);
	//show('level_label',itemized_value==1);
	enable('assembly_and_install_cost_input',itemized_value!=2);
	show('assembly_and_install_cost_total',itemized_value==1);
	show('assem_level_total_label',itemized_value==1);
	show('assembly_and_install_cost_input',(itemized_value==0 || itemized_value==2));
	show('assem_user_input_label',(itemized_value==0 || itemized_value==2));
	//show('level_label',itemized_value==1);
	//show('level_line',itemized_value==1);
	if (itemized_value==1)
	{
		detailed_cost_mec_device();
		detailed_cost_substructures();
		detailed_cost_offshore_infrastructure();
	}
	/*if (value('itemized_check_eng')==1)
	{
		enable_itemized_level_eng();
	}*/
	refresh();
}
