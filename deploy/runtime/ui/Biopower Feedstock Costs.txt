equations{ 'biopwr.feedstockcost.feedstock2_resource' } = define() {
	return ${biopwr.feedstock.feedstock2_resource}; };

equations{ 'biopwr.feedstockcost.subbit_resource' } = define() {
	return ${biopwr.feedstock.subbit_resource}; };

equations{ 'biopwr.feedstockcost.herb_resource' } = define() {
	return ${biopwr.feedstock.herb_resource}*${biopwr.feedstock.herb_obtainable}/100; };

equations{ 'biopwr.feedstockcost.rice_resource' } = define() {
	return ${biopwr.feedstock.rice_resource}*${biopwr.feedstock.rice_obtainable}/100; };

equations{ 'biopwr.feedstockcost.coal_per_mmbtu' } = define() {
	return ?${biopwr.feedstock.coal_opt}[0,((${biopwr.feedstock.bit_coal_frac}*${biopwr.feedstockcost.bit_price})+
(${biopwr.feedstock.subbit_coal_frac}*${biopwr.feedstockcost.subbit_price})+
(${biopwr.feedstock.lig_coal_frac}*${biopwr.feedstockcost.lig_price}))/2000/${biopwr.feedstock.total_coal_hhv}*1000000]; };

equations{ 'biopwr.feedstockcost.biomass_fuel_used' } = define() {
	return ${biopwr.feedstock.total_biomass}; };

equations{ 'biopwr.feedstockcost.stover_resource' } = define() {
	return ${biopwr.feedstock.stover_resource}*${biopwr.feedstock.stover_obtainable}/100; };

equations{ 'biopwr.feedstockcost.lig_resource' } = define() {
	return ${biopwr.feedstock.lig_resource}; };

equations{ 'biopwr.feedstockcost.woody_resource' } = define() {
	return ${biopwr.feedstock.woody_resource}*${biopwr.feedstock.woody_obtainable}/100; };

equations{ 'biopwr.feedstockcost.bit_resource' } = define() {
	return ${biopwr.feedstock.bit_resource}; };

equations{ 'biopwr.feedstockcost.feedstock1_resource' } = define() {
	return ${biopwr.feedstock.feedstock1_resource}; };

equations{ 'biopwr.feedstockcost.wheat_resource' } = define() {
	return ${biopwr.feedstock.wheat_resource}*${biopwr.feedstock.wheat_obtainable}/100; };

equations{ 'biopwr.feedstockcost.urban_resource' } = define() {
	return ${biopwr.feedstock.urban_resource}*${biopwr.feedstock.urban_obtainable}/100; };

equations{ 'biopwr.feedstockcost.biomass_cost' } = define() {
	if (${biopwr.feedstock.total_biomass}==0)
		return 0;
	else
		return (((${biopwr.feedstock.bagasse_biomass_frac}*${biopwr.feedstockcost.bagasse_price})+ (${biopwr.feedstock.barley_biomass_frac}*${biopwr.feedstockcost.barley_price})+ (${biopwr.feedstock.stover_biomass_frac}*${biopwr.feedstockcost.stover_price})+ (${biopwr.feedstock.rice_biomass_frac}*${biopwr.feedstockcost.rice_price})+ (${biopwr.feedstock.wheat_biomass_frac}*${biopwr.feedstockcost.wheat_price})+ (${biopwr.feedstock.forest_biomass_frac}*${biopwr.feedstockcost.forest_price})+ (${biopwr.feedstock.mill_biomass_frac}*${biopwr.feedstockcost.mill_price})+ (${biopwr.feedstock.urban_biomass_frac}*${biopwr.feedstockcost.urban_price})+ (${biopwr.feedstock.woody_biomass_frac}*${biopwr.feedstockcost.woody_price})+ (${biopwr.feedstock.herb_biomass_frac}*${biopwr.feedstockcost.herb_price})+ (${biopwr.feedstock.feedstock1_biomass_frac}*${biopwr.feedstockcost.feedstock1_price})+ (${biopwr.feedstock.feedstock2_biomass_frac}*${biopwr.feedstockcost.feedstock2_price}))+ (${biopwr.feedstockcost.fixed_delivery_cost}+(${biopwr.feedstock.collection_radius}*${biopwr.feedstockcost.var_delivery_cost}))) /2000/${biopwr.feedstock.total_biomass_hhv}*1000000;
};

equations{ 'biopwr.feedstockcost.bagasse_resource' } = define() {
	return ${biopwr.feedstock.bagasse_resource}*${biopwr.feedstock.bagasse_obtainable}/100; };

equations{ 'biopwr.feedstockcost.barley_resource' } = define() {
	return ${biopwr.feedstock.barley_resource}*${biopwr.feedstock.barley_obtainable}/100; };

equations{ 'biopwr.feedstockcost.biomass_fuel_cost' } = define() {
	return ${biopwr.feedstockcost.biomass_cost}/1000000*${biopwr.feedstock.total_biomass_hhv}*2000; };

equations{ 'biopwr.feedstockcost.coal_fuel_cost' } = define() {
	return ${biopwr.feedstockcost.coal_per_mmbtu}/1000000*${biopwr.feedstock.total_coal_hhv}*2000; };

	return ${biopwr.feedstockcost.biomass_fuel_cost}*(1-(${biopwr.feedstock.total_biomass_moisture}/100)); };
