on_load{'Marine Costs - Device Costs'} = define() {
	
	//Structural assembly
	//show('struct_level_total_label',0);
	show('structural_assembly_cost_total',value('structural_assembly_cost_method')==3);
	show('structural_assembly_cost_input',value('structural_assembly_cost_method')!=3);
	enable('structural_assembly_cost_input',value('structural_assembly_cost_method')!=2);

	
	//PTO
	//show('power_takeoff_system_total_label',0);
	show('power_takeoff_system_cost_total',value('power_takeoff_system_cost_method')==3);
	show('power_takeoff_system_cost_input',value('power_takeoff_system_cost_method')!=3);
	enable('power_takeoff_system_cost_input',value('power_takeoff_system_cost_method')!=2);
	
	
	//Mooring
	//show('mooring_cost_label',0);
	show('mooring_found_substruc_cost_total',value('mooring_found_substruc_cost_method')==3);
	show('mooring_found_substruc_cost_input',value('mooring_found_substruc_cost_method')!=3);
	enable('mooring_found_substruc_cost_input',value('mooring_found_substruc_cost_method')!=2);
	
	on_change{'device_cost_override'};
	override_device_cost();
	
	refresh();
};

on_change{'device_cost_override'} = define() {
	override_device_cost();
};

function override_device_cost()
{
	override_value = value('device_cost_override');
	enable('device_costs_total',override_value!=1);
	enable('device_cost_input',override_value==1);
	show('device_cost_input',override_value==1);
	show('device_costs_total',override_value!=1);
	
	
	//Structural Assembly Inputs
	enable('structural_assembly_cost_method',override_value!=1);
	if(value('structural_assembly_cost_method') < 2) {
		enable('structural_assembly_cost_input',override_value==0);
	}
	//enable('structural_assembly_cost_modeled',override_value==0);
	//enable('structural_assembly_cost_total',override_value==0);
	enable('structural_assembly_cost',override_value!=1);
	//show('struct_override_label',override_value==1);
	
	//PTO Inputs
	enable('power_takeoff_system_cost_method',override_value!=1);
	if(value('power_takeoff_system_cost_method') < 2) {
		enable('power_takeoff_system_cost_input',override_value==0);
	}
	//enable('power_takeoff_system_cost_modeled',override_value==0);
	//enable('power_takeoff_system_cost_total',override_value==0);
	enable('power_takeoff_system_cost',override_value!=1);
	//show('pto_override_label',override_value==1);
	
	//Mooring Inputs
	enable('mooring_found_substruc_cost_method',override_value!=1);
	if(value('mooring_found_substruc_cost_method') < 2) {
		enable('mooring_found_substruc_cost_input',override_value==0);
	}
	//enable('mooring_found_substruc_cost_modeled',override_value!=1);
	//enable('mooring_found_substruc_cost_total',override_value!=1);
	enable('mooring_found_substruc_cost',override_value!=1);
	//show('mooring_override_label',override_value==1);

	refresh();
}
