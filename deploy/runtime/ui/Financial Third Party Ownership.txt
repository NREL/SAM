function lease_or_ppa()
{
	lease = (value('lease_or_ppa') == 0); //0=lease, 1=ppa
	enable('lease_price', lease);
	enable('lease_escalation', lease);
	enable('ppa_price', !lease);
	enable('ppa_escalation', !lease);	
}

	lease_or_ppa();
};

	lease_or_ppa();
