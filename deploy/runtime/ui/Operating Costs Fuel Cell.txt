equations{ 'om_fuel_cost'} = define(){
	fuel_price = ${fc_fuel_cost};
	// need all prices in $/MMBtu 
	// Convert $/MCF to $/MMBtu (1 MCF = 1.037 MCF)
	if (${om_fuel_price_units} == 0)
	{
		for(i=0;i<#fuel_price;i++)
		{
			fuel_price[i] = fuel_price[i] * 1.037;
		}	
	}
	return fuel_price;
};

on_load{'Operating Costs Fuel Cell'} = define()
{
	enable('om_batt_fixed_cost',${en_batt});
	enable('om_batt_capacity_cost',${en_batt});
	enable('om_batt_variable_cost',${en_batt});
	enable('om_batt_replacement_cost',${en_batt});
	msg = '';
	if ( ${en_batt} == 0 )
	{
		msg = 'To enable battery cost inputs, choose Enable Battery on the Battery Storage page.';
	}
	property('msg_om_batt','Caption',msg);
	property('msg_om_batt','TextColour','Black');
	check_battery_om();
};

on_change{'om_batt_fixed_cost'} = define()
{
	if ( sum(${om_batt_fixed_cost}) > 0 && ${en_batt} == 0 ) 
	{
		msgbox('Battery not enabled.\nBattery must be enabled to specify battery operating costs.\n\nEither set the battery  cost to zero, or enable the battery on the Battery Storage page.');
		value('om_batt_fixed_cost',[0]);
	}
};

// also call this function on Battery None form in
// case user disables battery while battery O&M costs
// are not zero
function check_battery_om()
{

	is_batt_om = ( (sum(${om_batt_fixed_cost}) > 0) ||
				   (sum(${om_batt_capacity_cost}) > 0) ||
				   (sum(${om_batt_variable_cost}) > 0) ||
				   (sum(${om_batt_replacement_cost}) > 0) );

	if ( is_batt_om && ( ${en_batt} == 0) )
	{
		msgbox( 'Battery operating costs with no battery.\nFor a fuel cell with no battery, fuel cell costs must be set to zero.\n\nSetting battery operating costs to zero.' );
		value('om_batt_fixed_cost',[0]);
		value('om_batt_capacity_cost',[0]);
		value('om_batt_variable_cost',[0]);
		value('om_batt_replacement_cost',[0]);
	}
}
