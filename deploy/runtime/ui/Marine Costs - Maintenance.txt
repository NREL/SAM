		${other_scheduled_maintenance_cost};

equations {'scheduled_maintenance_cost_total_check'} = define() {
	if (${detailed_cost_choice_scheduled_maintenance}==0)
	{
		return ${scheduled_maintenance_cost};
	}
	else
	{
		return ${scheduled_maintenance_cost_total};
	}
};

};

			];

				${maintenance_cost_total}
			];
};

on_load{'Marine Costs - Maintenance'} = define() {
	on_change{'detailed_cost_choice_scheduled_maintenance'};
	detailed_cost_scheduled_maintenance();
	on_change{'detailed_cost_choice_unscheduled_maintenance'};
	detailed_cost_unscheduled_maintenance();
	on_change{'itemized_check_maintenance'};
	enable_itemized_maintenance();
	on_change{'maintenance_override_check'};
	check_override_maintenance();
};

on_change{'maintenance_override_check'} = define() {
	check_override_maintenance();
};

function check_override_maintenance() 
{
	check_value = value('maintenance_override_check');
	enable('maintenance_cost_method',check_value!=1);
	enable('maintenance_cost',check_value!=1);
	show('maintenance_override_label',check_value==1);
	
	refresh();
}

on_change{'detailed_cost_choice_scheduled_maintenance'} = define() {
	detailed_cost_scheduled_maintenance();
};

function detailed_cost_scheduled_maintenance()
{
	if (value('itemized_check_maintenance')==1)
	{
		detailed_value = value('detailed_cost_choice_scheduled_maintenance');
		enable('scheduled_maintenance_cost',detailed_value==0);
		show('scheduled_maintenance_cost',detailed_value==0);
		enable('scheduled_maintenance_cost_total',detailed_value);
		show('scheduled_maintenance_cost_total',detailed_value);
		enable('marine_energy_converter_scheduled_maintenance_cost',detailed_value);
		enable('balance_of_system_scheduled_maintenance_cost',detailed_value);
		enable('other_scheduled_maintenance_cost',detailed_value);
		
		refresh();
	}
}

on_change{'detailed_cost_choice_unscheduled_maintenance'} = define() {
	detailed_cost_unscheduled_maintenance();
};

function detailed_cost_unscheduled_maintenance()
{
	if (value('itemized_check_maintenance')==1)
	{
		detailed_value = value('detailed_cost_choice_unscheduled_maintenance');
		enable('unscheduled_maintenance_cost',detailed_value==0);
		show('unscheduled_maintenance_cost',detailed_value==0);
		enable('unscheduled_maintenance_cost_total',detailed_value);
		show('unscheduled_maintenance_cost_total',detailed_value);
		enable('marine_energy_converter_unscheduled_maintenance_cost',detailed_value);
		enable('balance_of_system_unscheduled_maintenance_cost',detailed_value);
		enable('unscheduled_maintenance_contingency_cost',detailed_value);
		enable('other_unscheduled_maintenance_cost',detailed_value);
		
		refresh();
	}
}

on_change{'itemized_check_maintenance'} = define() {
	enable_itemized_maintenance();
};

function enable_itemized_maintenance()
{
	itemized_value = value('itemized_check_maintenance');
	enable('long_term_service_agreement_cost',itemized_value==1);
	enable('scheduled_maintenance_cost',itemized_value==1);
	enable('scheduled_maintenance_cost_total',itemized_value==1);
	enable('detailed_cost_choice_scheduled_maintenance',itemized_value==1);
	enable('marine_energy_converter_scheduled_maintenance_cost',itemized_value==1);
	enable('balance_of_system_scheduled_maintenance_cost',itemized_value==1);
	enable('other_scheduled_maintenance_cost',itemized_value==1);
	enable('unscheduled_maintenance_cost',itemized_value==1);
	enable('unscheduled_maintenance_cost_total',itemized_value==1);
	enable('detailed_cost_choice_unscheduled_maintenance',itemized_value==1);
	enable('marine_energy_converter_unscheduled_maintenance_cost',itemized_value==1);
	enable('balance_of_system_unscheduled_maintenance_cost',itemized_value==1);
	enable('unscheduled_maintenance_contingency_cost',itemized_value==1);
	enable('other_unscheduled_maintenance_cost',itemized_value==1);
	enable('other_maintenance_cost',itemized_value==1);
	enable('maintenance_cost_input',itemized_value!=2);
	show('maintenance_cost_total',itemized_value==1);
	show('maintenance_level_total_label',itemized_value==1);
	show('maintenance_cost_input',(itemized_value==0 || itemized_value==2));
	if (itemized_value==1)
	{
		detailed_cost_scheduled_maintenance();
		detailed_cost_unscheduled_maintenance();
	}
	refresh();
}
