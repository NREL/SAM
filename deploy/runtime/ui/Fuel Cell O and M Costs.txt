equations{ 'om_fuel_cost'} = define(){
	fuel_price = ${fc_fuel_cost};
	// need all prices in $/MMBtu 
	// Convert $/MCF to $/MMBtu (1 MCF = 1.037 MCF)
	if (${om_fuel_price_units} == 0)
		for(i=0;i<#fuel_price;i++)
			fuel_price[i] = fuel_price[i] * 1.037;
	
	return fuel_price;
};
