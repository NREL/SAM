equations{'ui_total_capacity'} = define() {

equations{'ui_total_unsubscribed_kw'} = define() {
	return ${system_capacity}
	- ( ${ui_share_kw_1}
	+ ${ui_share_kw_2}
	+ ${ui_share_kw_3}
	+ ${ui_share_kw_4} );
};

equations{'ui_total_unsubscribed_share'} = define() {
	return 100 - 
	( ${subscriber1_share}[0]
	+ ${subscriber2_share}[0]
	+ ${subscriber3_share}[0]
	+ ${subscriber4_share}[0] );
};

equations{'ui_share_kw_1'} = define() {
	return ${subscriber1_share}[0]/100 * ${system_capacity};
};

	return ${subscriber3_share}[0]/100 * ${system_capacity};
};

equations{'ui_share_kw_4'} = define() {
	return ${subscriber4_share}[0]/100 * ${system_capacity};

function enable_growth_rate() 
{
	enable('subscriber1_growth', #${subscriber1_share} == 1 ); 
	enable('subscriber2_growth', #${subscriber2_share} == 1 ); 
	enable('subscriber3_growth', #${subscriber3_share} == 1 ); 
	enable('subscriber4_growth', #${subscriber4_share} == 1 ); 
}

function enable_retail_rate_escalation()
{
	enable('subscriber1_retail_rate_escal', #${subscriber1_retail_rate} == 1 );
	enable('subscriber2_retail_rate_escal', #${subscriber2_retail_rate} == 1 );
	enable('subscriber3_retail_rate_escal', #${subscriber3_retail_rate} == 1 );
	enable('subscriber4_retail_rate_escal', #${subscriber4_retail_rate} == 1 );
}

function subscriber_share_message()
{
	if ( ${ui_total_unsubscribed_share} < 0 )
	{
		msg = 'The project is oversubscribed by ' + -${ui_total_unsubscribed_share} + '% in Year 1! Reduce the anchor, general, or subsidized subscriber subscription rate.'; 
		color = 'red';
	}
	else
	{
		msg = 'See the unsubscribed share by year in the results after running a simulation.';
		color = 'black';
	}
	property('msg_subscriber_share','Caption',msg);
	property('msg_subscriber_share','TextColour',color);
}

function enable_retail_rate()
{
	enable('subscriber1_retail_rate',${enable_subscriber_retail_rate});
	enable('subscriber2_retail_rate',${enable_subscriber_retail_rate});
	enable('subscriber3_retail_rate',${enable_subscriber_retail_rate});
	enable('subscriber4_retail_rate',${enable_subscriber_retail_rate});
	enable('subscriber1_retail_rate_escal',${enable_subscriber_retail_rate});
	enable('subscriber2_retail_rate_escal',${enable_subscriber_retail_rate});
	enable('subscriber3_retail_rate_escal',${enable_subscriber_retail_rate});
	enable('subscriber4_retail_rate_escal',${enable_subscriber_retail_rate});
	if ( ${enable_subscriber_retail_rate} )
	{
		enable_retail_rate_escalation();
	}
}
