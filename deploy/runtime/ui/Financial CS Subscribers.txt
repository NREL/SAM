equations{'ui_total_capacity'} = define() {
	return ${system_capacity};
};

equations{'ui_total_unsubscribed_kw'} = define() {
	return ${system_capacity}
	- ( ${ui_share_kw_1}
	+ ${ui_share_kw_2}
	+ ${ui_share_kw_3}
	+ ${ui_share_kw_4} );
};

equations{'ui_total_unsubscribed_share'} = define() {
	return 100 - 
	( ${subscriber1_share}[0]
	+ ${subscriber2_share}[0]
	+ ${subscriber3_share}[0]
	+ ${subscriber4_share}[0] );
};

equations{'ui_share_kw_1'} = define() {
	return ${subscriber1_share}[0]/100 * ${system_capacity};
};

equations{'ui_share_kw_2'} = define() {
	return ${subscriber2_share}[0]/100 * ${system_capacity};
};

equations{'ui_share_kw_3'} = define() {
	return ${subscriber3_share}[0]/100 * ${system_capacity};
};

equations{'ui_share_kw_4'} = define() {
	return ${subscriber4_share}[0]/100 * ${system_capacity};
};

on_load{'Financial CS Subscribers'} = define()
{
	property('subscriber1_share','FixedLength',${analysis_period});
	property('subscriber2_share','FixedLength',${analysis_period});
	property('subscriber3_share','FixedLength',${analysis_period});
	property('subscriber4_share','FixedLength',${analysis_period});
	enable('subscriber1_growth', #${subscriber1_share} == 1 ); 
	enable('subscriber2_growth', #${subscriber2_share} == 1 ); 
	enable('subscriber3_growth', #${subscriber3_share} == 1 ); 
	enable('subscriber4_growth', #${subscriber4_share} == 1 ); 
	subscriber_share_message();
};

on_change{'subscriber1_share'} = define()
{
	enable('subscriber1_growth', #${subscriber1_share} == 1 );
	subscriber_share_message();
};
on_change{'subscriber2_share'} = define()
{
	enable('subscriber2_growth', #${subscriber2_share} == 1 );
	subscriber_share_message();
};
on_change{'subscriber3_share'} = define()
{
	enable('subscriber3_growth', #${subscriber3_share} == 1 );
	subscriber_share_message();
};
on_change{'subscriber4_share'} = define()
{
	enable('subscriber4_growth', #${subscriber4_share} == 1 );
	subscriber_share_message();
};

function subscriber_share_message()
{
	if ( ${ui_total_unsubscribed_share} < 0 )
	{
		msg = 'The project is oversubscribed by ' + -${ui_total_unsubscribed_share} + '% in Year 1! Reduce the subscriber share for one or more subscribers.'; 
		color = 'red';
	}
	else
	{
		msg = 'See the unsubscribed share by year in the results after running a simulation.';
		color = 'black';
	}
	property('msg_subscriber_share','Caption',msg);
	property('msg_subscriber_share','TextColour',color);
}
