equations {'balance_of_system_cost_total'} = define() {
	if (${balance_of_system_cost_override}!=1)
	{
		return ${development_cost}+${eng_and_mgmt_cost}+${elec_infras_cost}
		+${site_access_port_staging_cost}+${plant_commissioning_cost}+
		${assembly_and_install_cost}+${other_infrastructure_cost}; //+elec_infrac
	}
	else
	{
		return ${balance_of_system_cost_input};
	}
};

on_load{'Marine Costs - BoS'} = define() {
	on_change{'balance_of_system_cost_override'};
	override_balance_of_system_cost();
	on_change{'plant_commissioning_cost_override'};
	
	//Development
	show('development_cost_total',value('development_cost_method')==3);
	show('development_cost_input',value('development_cost_method')!=3);
	enable('development_cost_input',value('development_cost_method')!=2);
	
	//E&M
	show('eng_and_mgmt_cost_total',value('eng_and_mgmt_cost_method')==3);
	show('eng_and_mgmt_cost_input',value('eng_and_mgmt_cost_method')!=3);

	show('elec_infras_cost_input',value('elec_infras_cost_method')!=3);
	enable('elec_infras_cost_input',value('elec_infras_cost_method')!=2);

	enable('plant_commissioning_cost_input',value('plant_commissioning_cost_method')!=2);
	
	//Site access, Port and Staging
	show('site_access_port_staging_cost_total',value('site_access_port_staging_cost_method')==3);
	show('site_access_port_staging_cost_input',value('site_access_port_staging_cost_method')!=3);

	enable('other_infrastructure_cost_input',value('other_infrastructure_cost_method')!=2);
	
	refresh();

function override_balance_of_system_cost()
{
	override_value = value('balance_of_system_cost_override');
	enable('balance_of_system_cost_total',override_value!=1);
	enable('balance_of_system_cost_input',override_value==1);
	show('balance_of_system_cost_input',override_value==1);
	show('balance_of_system_cost_total',override_value!=1);
	
	//Development Inputs
	enable('development_cost_method',override_value!=1);
	//enable('development_cost_input',override_value==0);
	//enable('development_cost_modeled',override_value==0);
	//enable('development_cost_total',override_value==0);
	enable('development_cost',override_value!=1);
	//show('development_override_label',override_value==1);
	
	//E&M Inputs
	enable('eng_and_mgmt_cost_method',override_value==0);
	//enable('eng_and_mgmt_cost_input',override_value==0);
	//enable('eng_and_mgmt_cost_modeled',override_value==0);
	//enable('eng_and_mgmt_cost_total',override_value==0);
	enable('eng_and_mgmt_cost',override_value==0);
	//show('eng_mgmt_override_label',override_value==1);
	
	//Electrical Infrastructure Inputs
	enable('elec_infras_cost_method',override_value==0);
	//enable('elec_infras_cost_input',override_value==0);
	//enable('elec_infras_cost_modeled',override_value==0);
	//enable('elec_infras_cost_total',override_value==0);
	enable('elec_infras_cost',override_value==0);
	//show('elec_infras_override_label',override_value==1);
	
	//Plant Commissioning Inputs
	//plant_value = value('override_plant');
	enable('plant_commissioning_cost_method',override_value!=1);
	//enable('plant_commissioning_cost_input',override_value==0);
	//enable('plant_commissioning_cost_modeled',override_value==0);
	enable('plant_commissioning_cost',override_value!=1);
	//show('plant_override_label',plant_value!=0);
	
	//Site Access Inputs
	enable('site_access_port_staging_cost_method',override_value!=1);
	//enable('site_access_port_staging_cost_input',override_value==0);
	//enable('site_access_port_staging_cost_modeled',override_value==0);
	//enable('site_access_port_staging_cost_total',override_value==0);
	enable('site_access_port_staging_cost',override_value!=1);
	//show('site_override_label',override_value==1);
	
	//Assembly and Installation Inputs
	enable('assembly_and_install_cost_method',override_value!=1);
	//enable('assembly_and_install_cost_input',override_value==0);
	//enable('assembly_and_install_cost_modeled',override_value==0);
	//enable('assembly_and_install_cost_total',override_value==0);
	enable('assembly_and_install_cost',override_value!=1);
	//show('assembly_override_label',override_value==1);
	
	//Other Infrastructure Inputs
	enable('other_infrastructure_cost_method',override_value!=1);
	//enable('other_infrastructure_cost_input',override_value==0);
	//enable('other_infrastructure_cost_modeled',override_value==0);
	//enable('other_infrastructure_cost_total',override_value==0);
	enable('other_infrastructure_cost',override_value!=1);
	//show('other_infras_override_label',override_value==1);
	
	refresh();
}
