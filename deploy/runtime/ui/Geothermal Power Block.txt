function update_hybrid_dispatch_inputs() {
	bEnable = (${model_choice}==2 && ${geopowerblock.pwrb.condenser_type} == 2);
	
	enable("hc_ctl1", bEnable);
	enable("hc_ctl2", bEnable);
	enable("hc_ctl3", bEnable);
	enable("hc_ctl4", bEnable);
	enable("hc_ctl5", bEnable);
	enable("hc_ctl6", bEnable);
	enable("hc_ctl7", bEnable);
	enable("hc_ctl8", bEnable);
	enable("hc_ctl9", bEnable);
}
on_load{'Geothermal Power Block'} = define() {
	update_hybrid_dispatch_inputs();
};

on_change{'geopowerblock.pwrb.condenser_type'} = define() {
	update_hybrid_dispatch_inputs();
};

on_change{'model_choice'} = define() {
	update_hybrid_dispatch_inputs();
};
