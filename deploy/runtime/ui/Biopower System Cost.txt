equations{ 'biopwr.cost.sales_tax.value' } = define() {
	return ${sales_tax_rate}; };

equations{ 'biopwr.cost.dryer_capacity' } = define() {
	return ?${biopwr.plant.drying_method}[0,0,${biopwr.plant.nameplate}];
};

equations{ 'biopwr.cost.total_direct' } = define() {
	return ${biopwr.cost.contingency}+${biopwr.cost.boiler}+${biopwr.cost.turbine}+${biopwr.cost.prep}+${biopwr.cost.dryer}+${biopwr.cost.equipment}+${biopwr.cost.bop}; };

equations{ 'biopwr.cost.bop_capacity' } = define() {
	return ${biopwr.plant.nameplate}-${biopwr.plant.par}; };

equations{ 'biopwr.cost.epc.total' } = define() {
	return ${biopwr.cost.epc.fixed}+${biopwr.cost.epc.nonfixed}; };

equations{ 'biopwr.cost.total_indirect' } = define() {
	return ${biopwr.cost.epc.total}+${biopwr.cost.plm.total}; };

equations{ 'biopwr.cost.turbine' } = define() {
	return ${biopwr.cost.turbine_capacity}*${biopwr.cost.turbine_per_kw}; };

equations{ 'biopwr.cost.equipment' } = define() {
	return ${biopwr.cost.equipment_capacity}*${biopwr.cost.equipment.cost_per_kw}; };

equations{ 'biopwr.cost.cap_per_boiler' } = define() {
	return ${biopwr.plant.boiler.cap_per_boiler}; };

equations{ 'biopwr.cost.contingency' } = define() {
	return ${biopwr.cost.contingency_percent}/100*(${biopwr.cost.boiler}+${biopwr.cost.turbine}+${biopwr.cost.prep}+${biopwr.cost.equipment}+${biopwr.cost.bop}); };

equations{ 'biopwr.cost.plm.total' } = define() {
	return ${biopwr.cost.plm.fixed}+${biopwr.cost.plm.nonfixed}; };

equations{ 'biopwr.cost.installed_per_capacity' } = define() {
	return ${total_installed_cost}/${biopwr.plant.nameplate}; };

equations{ 'biopwr.cost.boiler_capacity' } = define() {
	return ${biopwr.plant.nameplate}; };

equations{ 'biopwr.cost.dryer' } = define() {
	return ${biopwr.cost.dryer_capacity}*${biopwr.cost.dryer_per_kw}; };

equations{ 'biopwr.cost.prep' } = define() {
	return ${biopwr.cost.prep_capacity}*${biopwr.cost.prep_per_cap}; };

equations{ 'biopwr.cost.boiler' } = define() {
	return ${biopwr.cost.boiler_capacity}*${biopwr.cost.boiler.cost_per_kw}; };

equations{ 'biopwr.cost.epc.nonfixed' } = define() {
	return ${biopwr.cost.total_direct}*${biopwr.cost.epc.percent}/100; };

equations{ 'total_installed_cost' } = define() {
	return ${biopwr.cost.total_direct}+${biopwr.cost.total_indirect}+${biopwr.cost.sales_tax.total}; };

equations{ 'biopwr.cost.bop' } = define() {
	return ${biopwr.cost.bop_per_kw}*${biopwr.cost.bop_capacity}; };

equations{ 'biopwr.cost.turbine_capacity' } = define() {
	return ${biopwr.plant.nameplate}; };

equations{ 'biopwr.cost.equipment_capacity' } = define() {
	return ${biopwr.plant.nameplate}; };

equations{ 'biopwr.cost.plm.nonfixed' } = define() {
	return ${biopwr.cost.plm.percent}/100*${biopwr.cost.total_direct}; };

equations{ 'biopwr.cost.sales_tax.total' } = define() {
	return ${biopwr.cost.sales_tax.value}/100*${biopwr.cost.sales_tax.percent}/100*${biopwr.cost.total_direct}; };

equations{ 'biopwr.cost.prep_capacity' } = define() {
	return ${biopwr.plant.nameplate}; };
	
equations { 'total_direct_cost' } = define () {
	return ${biopwr.cost.total_direct};
};

equations { 'total_indirect_cost' } = define () {
	return ${biopwr.cost.total_indirect};
};

equations { 'system_use_recapitalization' } = define () {
	return 0;};
