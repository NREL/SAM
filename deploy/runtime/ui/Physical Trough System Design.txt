equations{ '$MIMO$ Physical_Trough_System_Design_Equations' } = define() {

	// csp.dtr.pwrb.nameplate
	data.P_ref = ${P_ref};
	data.gross_net_conversion_factor = ${gross_net_conversion_factor};

	// q_pb_design
	data.eta_ref = ${eta_ref};

	success = ssc_eqn("Physical_Trough_System_Design_Equations", data);
	if (!success) {
		${csp.dtr.pwrb.nameplate} = -999.;
		${q_pb_design} = -999.;
	}
	else {
		${csp.dtr.pwrb.nameplate} = data.csp_dtr_pwrb_nameplate;
		${q_pb_design} = data.q_pb_design;
	}
	
	return 0;
};
