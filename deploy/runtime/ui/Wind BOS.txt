equations{ 'bos_total_cost_per_kw' } = define() {
	return ${bos_total_budgeted_cost}/(${farm_size_MW}*1000);
};

equations{ 'construction_time' } = define() {
	if ( ${construction_time_choice} == 0 )
		return ${construction_time_input};
	else
		return floor(0.5+(0.0001*${wind_farm_num_turbines}*${wind_farm_num_turbines} + 0.0963*${wind_farm_num_turbines} + 2.7432));
};

equations{ 'construction_time_sugg' } = define() {
	return floor(0.5+(0.0001*${wind_farm_num_turbines}*${wind_farm_num_turbines} + 0.0963*${wind_farm_num_turbines} + 2.7432));
};

equations{ 'om_building_size' } = define() {
	// O&M Building Size (ft2)
	if ( ${om_building_size_choice} == 0 )
		return ${om_building_size_input};
	else if (${farm_size_MW} < 200)
		return 3000;
	else if (${farm_size_MW} < 500)
		return 5000;
	else if (${farm_size_MW} < 800)
		return 7000;
	else if (${farm_size_MW}< 1000)
		return 9000;
	else
		return 12000;
};

equations{ 'om_building_size_sugg' } = define() {
	// O&M Building Size (ft2)
	if (${farm_size_MW} < 200)
		return 3000;
	else if (${farm_size_MW} < 500)
		return 5000;
	else if (${farm_size_MW} < 800)
		return 7000;
	else if (${farm_size_MW}< 1000)
		return 9000;
	else
		return 12000;
};

equations{ 'quantity_test_met_towers' } = define() {
	if ( ${quantity_test_met_towers_choice} == 0 )
		return ${quantity_test_met_towers_input};
	else
		return floor(0.5+(${farm_size_MW} / 75.0));
};

equations{ 'quantity_test_met_towers_sugg' } = define() {
	return floor(0.5+(${farm_size_MW} / 75.0));
};

equations{ 'quantity_permanent_met_towers' } = define() {
	if ( ${quantity_permanent_met_towers_choice} == 0 )
		return ${quantity_permanent_met_towers_input};
	else if (${farm_size_MW} < 100)
		return 1;
	else if (${farm_size_MW}< 200)
		return 2;
	else
		return floor(${farm_size_MW} / 100.0);
};

equations{ 'quantity_permanent_met_towers_sugg' } = define() {
	if (${farm_size_MW} < 100)
		return 1;
	else if (${farm_size_MW}< 200)
		return 2;
	else
		return floor(${farm_size_MW} / 100.0);
};

equations{ 'weather_delay_days' } = define() {

on_change{"btn_bos_apply"} = define()
{
	install_type = value('install_type');
	if (install_type == 1)
	{
		msgbox('This Balance of System cost model is only meant to be applied to land-based installations.');
		return;
	}
	value('bos_cost_fixed', 0);
	value('bos_cost_per_kw', value('bos_total_cost_per_kw'));
	value('bos_cost_per_turbine', 0);
	value('sales_tax_basis', 0);
	msgbox('The total cost estimate of $' + value('bos_total_cost_per_kw') +
		'/kW was applied to the Balance of System Cost per kW input. ' +
		'The Balance of System Fixed cost and Cost per turbine inputs were zeroed.');
};
