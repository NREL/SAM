	tech = technology();

	{ return ${batt_computed_bank_capacity}; }

on_load{'PV Battery O and M Costs'} = define() {
	tech = technology();
	generic = (tech == "Generic Battery");
	if (generic) {
		property('pv_label','Caption','Power Plant');
	}
	else {
		property('pv_label','Caption', 'PV');
	}
	refresh();
};
