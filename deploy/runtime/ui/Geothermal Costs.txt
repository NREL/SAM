equations{ 'geotherm.cost.pump_per_pump' } = define() {
	return ${geotherm.cost.pump_per_hp} * sqrt(${geotherm.cost.pump_size}); };

equations{ 'geotherm.cost.plm.total' } = define() {
	return ${geotherm.cost.plm.fixed}+${geotherm.cost.plm.nonfixed}; };

equations{ 'geotherm.cost.prod_drill' } = define() {
	return  ${geotherm.cost.prod_per_well} * ${geotherm.cost.prod_num_wells}; };

equations{ 'geotherm.cost.prod_req' } = define() {
	return ${num_wells_getem}; };

equations{ 'geotherm.cost.pumping.amount' } = define() {
	return  ?${geotherm.cost.pumping.calc}[
${geotherm.cost.pumping.amount_specified},${geotherm.cost.pumps_total}]; };

equations{ 'geotherm.cost.prod_inj_drill' } = define() {
	return  ${geotherm.cost.prod_drill} + ${geotherm.cost.inj_drill}; };

equations{ 'geotherm.cost.expl_drill' } = define() {
	return  ${geotherm.cost.expl_per_well} * ${geotherm.cost.expl_num_wells}; };

equations{ 'geotherm.cost.expl_total' } = define() {
	return  ${geotherm.cost.expl_drill} + ${geotherm.cost.expl_non_drill}; };

equations{ 'geotherm.cost.conf_per_well' } = define() {
	return  ${geotherm.cost.conf_multiplier} * ${geotherm.cost.prod_per_well}; };

equations{ 'geotherm.cost.prod_wells_drilled' } = define() {
	return ( ${geotherm.cost.prod_req} - ${geotherm.cost.confirm_wells_num} ); };

equations{ 'geotherm.cost.pump_total_per_pump' } = define() {
	return ${geotherm.cost.pump_installation} + ${geotherm.cost.pump_per_pump}; };

equations{ 'total_indirect_cost' } = define() {
	return  ?${geotherm.cost.indirect.calc}[
${geotherm.cost.indirect.amount_specified},${geotherm.cost.epc.total}+${geotherm.cost.plm.total}+${geotherm.cost.sales_tax.total}]; };

equations{ 'geotherm.cost.pumps_total' } = define() {
	return ${geotherm.cost.pump_total_per_pump} * ${geotherm.cost.num_pumps}; };

equations{ 'total_direct_cost' } = define() {
	return ${geotherm.cost.contingency}+${geotherm.cost.capital_total}; };

equations{ 'geotherm.cost.drilling.amount' } = define() {
	return  ?${geotherm.cost.drilling.calc}[
${geotherm.cost.drilling.amount_specified}, /* =0 user specified */
${geotherm.cost.expl_total} + ${geotherm.cost.conf_total} + ${geotherm.cost.prod_inj_total} + ${geotherm.cost.surf_total} + ${geotherm.cost.stim_total}]; };

equations{ 'geotherm.cost.prod_inj_total' } = define() {
	return  ${geotherm.cost.prod_inj_drill} + ${geotherm.cost.prod_inj_non_drill}; };

equations{ 'geotherm.cost.recap' } = define() {
	return  ?${geotherm.cost.recap_use_calc}[
${geotherm.cost.recap_specified}, /* =0 user specified */
${geotherm.cost.conf_drill} /* =1 use calculated */
+ ${geotherm.cost.prod_inj_drill}
+ ${geotherm.cost.surf_total}
+ ${geotherm.cost.pumps_total}]; };

equations{ 'geotherm.cost.prod_num_wells' } = define() {
	return  ${geotherm.cost.prod_wells_drilled}; };

equations{ 'geotherm.cost.plm.nonfixed' } = define() {
	return ${total_direct_cost}*${geotherm.cost.plm.percent}/100; };

equations{ 'geotherm.cost.num_pumps' } = define() {
	return ${geotherm.cost.prod_req}; };

equations{ 'geotherm.cost.pump_size' } = define() {
	return ${pump_size_hp}; };

equations{ 'geotherm.cost.sales_tax.total' } = define() {
	return ${geotherm.cost.sales_tax.value}/100*${total_direct_cost}*${geotherm.cost.sales_tax.percent}/100; };

equations{ 'geotherm.cost.stim_num_wells' } = define() {
	return  ${geotherm.cost.prod_req} + ${geotherm.cost.inj_num_wells}; };

equations{ 'geotherm.cost.surf_total' } = define() {
	return  ${geotherm.cost.surf_non_drill}; };

equations{ 'geotherm.cost.installed_per_capacity' } = define() {
	return ${total_installed_cost}/${geotherm.net_output}/1000; };

function reset_geothermal_plant_cost()
{
	// This should never get called for flash plants - the checkbox that calls this code is disabled for flash plants
	if (value("conversion_type") == 0) // binary plant
	{
        if ( ( value("geotherm.cost.plant_auto_estimate") == 0 ) && ( value("geotherm.cost.plant_per_kW.bak") != 0 ) )
		{
            // The users original value will get saved into "geotherm.cost.plant_per_kW.bak" the first time they
            // check the box to automatically calculate costs.  This is done in mimo_geothermal_eval, in samwx\samexe\src\uievalspecial.cpp

			// When they uncheck the box, put back user's original value, and set the backup = 0 (otherwise it won't be written
            // over again when they re-check the box
            value("geotherm.cost.plant_per_kW", value("geotherm.cost.plant_per_kW.bak") );
            value("geotherm.cost.plant_per_kW.bak", 0 );

		}
	}
}


function update_cost_input_control()
{
	// show calculated value if selected
	show("geotherm.cost.plant_per_kW", value("geotherm.cost.plant_auto_estimate") );
	
	// show input value if not selected
	show("geotherm.cost.plant_per_kW_input", value("geotherm.cost.plant_auto_estimate") == 0);

}

	update_cost_input_control();	
};

	update_cost_input_control();
