equations{ 'engr_total' } = define() {
	return ${engr_percent}/100*${total_direct_cost}
			+ ${modulearray_power}*1000*${engr_per_watt} 
			+ ${battery_power} * ${engr_per_battkw}
			+ ${fuelcell_power_nameplate} * ${engr_per_fckw}
			+ ${engr_fixed};
};

equations{ 'inverter_num_units' } = define() {
	if( technology() == 'Flat Plate PV' ) return ${inverter_count};
	elseif ( technology() == 'PV Battery' ) return ${inverter_count};
	elseif ( technology() == 'PVWatts' ) return 1;
	elseif ( technology() == 'PVWatts Battery' ) return 1;
	elseif ( technology() == 'Fuel Cell') return 1;
	else return 0;
};

equations{ 'module_power' } = define() {
	tech = technology();
	if ( tech == 'PVWatts' || tech == 'PVWatts Battery' || tech == 'Fuel Cell') { 
		return ${system_capacity};
	}
	else {
		// Flat Plate PV
		return 0.001*?${module_model}
				[
					${spe_power},
					${cec_p_mp_ref},
					${6par_pmp},
					${snl_ref_pmp},
					${sd11par_Pmp0}
				];
	}
};

equations{ 'contingency' } = define() {
	return ${contingency_percent}/100*(
				 ${module_total}
				+${inverter_total}
				+${bos_equip_total}
				+${install_labor_total} 
				+${install_margin_total} );
};

equations{ 'inverter_power' } = define() {
	tech = technology();
	if ( tech == 'PVWatts' || tech == 'PVWatts Battery' || tech == 'Fuel Cell') {
		return ${system_capacity}/${dc_ac_ratio};
	}
	else return 0.001*?${inverter_model}
			[
				${inv_snl_paco},
				${inv_ds_paco},
				${inv_pd_paco},
				${inv_cec_cg_paco}
			];
};

equations{ 'sales_tax_value' } = define() {
	return ${sales_tax_rate}; };

equations{ 'modulearray_power' } = define() {
	return ${module_power}*${module_num_units}; };

equations{ 'module_total' } = define() {
	return ?${module_costunits}
		[
			(${module_num_units}*${module_power})*1000,
			${module_num_units}
		]
		*${per_module};
};

equations{ 'land_total' } = define() {
	return ${land_per_acre}*${land_area_value} 
		+ ${land_percent}/100*${total_direct_cost} 
		+ ${modulearray_power}*1000*${land_per_watt} 
		+ ${battery_power} * ${land_per_battkw}
		+ ${fuelcell_power_nameplate} * ${land_per_fckw}
		+ ${land_fixed}; };

equations{ 'permitting_total' } = define() {
	return ${permitting_percent}/100*${total_direct_cost} 
		+ ${modulearray_power}*1000*${permitting_per_watt} 
		+ ${battery_power} * ${permitting_per_battkw}
		+ ${fuelcell_power_nameplate} * ${permitting_per_fckw}
		+ ${permitting_fixed}; };

equations{ 'sales_tax_total' } = define() {
	return ${sales_tax_value}/100
		*${total_direct_cost}
		*${sales_tax_percent}/100; };

equations{ 'install_labor_total' } = define() {
	return ${install_labor_fixed}
		+ ${modulearray_power}*1000*${install_labor_perwatt}
		+ ${battery_power} * ${install_labor_battperkw}
		+ ${fuelcell_power_nameplate} * ${install_labor_fcperkw}
		/*+ ${modulearray_area}*${install_labor_perarea}*/; };

equations{ 'module_num_units' } = define() {
	tech = technology();
	if ( tech == 'PVWatts' || tech == 'PVWatts Battery' || tech == 'Fuel Cell') { 
		return 1;
	}
	else return ${total_modules};
};

equations{ 'landprep_total' } = define() {
	return  ${landprep_per_acre}*${land_area_value} 
		+ ${landprep_percent}/100*${total_direct_cost} 
		+ ${modulearray_power}*1000*${landprep_per_watt} 
		+ ${battery_power} * ${landprep_per_battkw}
		+ ${fuelcell_power_nameplate} * ${landprep_per_fckw}
		+ ${landprep_fixed}; };


equations{ 'install_margin_total' } = define() {
	return ${install_margin_fixed}
		+ ${modulearray_power}*1000*${install_margin_perwatt}
		+ ${battery_power} * ${install_margin_battperkw}
		+ ${fuelcell_power_nameplate} * ${install_margin_fcperkw}
		/*+ ${modulearray_area}*${install_margin_perarea}*/; };

equations{ 'total_installed_cost' } = define() {
	return ${total_direct_cost}+${total_indirect_cost}; };

equations{ 'inverterarray_power' } = define() {
	return ${inverter_power}*${inverter_num_units}; };

equations{ 'modulearray_area' } = define() {
	tech = technology();
	if ( tech == 'PVWatts' || tech == 'PVWatts Battery' || tech == 'Fuel Cell') {
		return 0;
	}
	else return ${total_land_area};
};

equations{ 'total_indirect_cost' } = define() {
	return ${permitting_total}
		+${engr_total}
		+${grid_total}
		+${land_total}
		+${landprep_total}
		+${sales_tax_total}; };

equations{ 'grid_total' } = define() {
	return ${grid_percent}/100*${total_direct_cost} 
		+ ${modulearray_power}*1000*${grid_per_watt} 
		+ ${battery_power} * ${grid_per_battkw}
		+ ${fuelcell_power_nameplate} * ${grid_per_fckw}
		+ ${grid_fixed}; };

equations{ 'inverter_total' } = define() {
	return ?${inverter_costunits}
				[
					(${inverter_num_units}*${inverter_power})*1000,
					(${module_num_units}*${module_power})*1000,
					${inverter_num_units}
				]
				*${per_inverter}; };


equations{ 'bos_equip_total' } = define() {
	return ${bos_equip_fixed}
+ ${modulearray_power}*1000*${bos_equip_perwatt}
+ ${battery_power} * ${bos_equip_battperkw}
+ ${fuelcell_power_nameplate} * ${bos_equip_fcperkw}
/*+ ${modulearray_area}*${bos_equip_perarea}*/; };


equations{ 'installed_per_capacity' } = define() {
	return 0.001*${total_installed_cost}/${modulearray_power}; };

equations{ 'subtotal_direct' } = define() {
	return ${module_total}
		+${inverter_total}
		+${battery_total}
		+${fuelcell_total}
		+${bos_equip_total}
		+${install_labor_total}
		+${install_margin_total}; };

equations{ 'total_direct_cost' } = define() {
	return ${module_total}
		+${inverter_total}
		+${battery_total}
		+${fuelcell_total}
		+${bos_equip_total}
		+${install_labor_total}
		+${install_margin_total}
		+${contingency}; };

equations { 'battery_total' } = define () {
	return ${battery_energy}*${battery_per_kWh} + ${battery_power}*${battery_per_kW};};
	
equations { 'battery_energy' } = define () {
	if( technology() == 'PV Battery' || technology() == 'Fuel Cell')
			return ?${en_batt}
				[0,	${batt_computed_bank_capacity}];
	else if ((technology() == 'PVWatts Battery' )) 
		{
			return ?${batt_simple_enable}
				[0,	${batt_simple_kwh}];
		}
	else
		return 0;
};
equations { 'battery_power' } = define () {
	if( technology() == 'PV Battery' || technology() == 'Fuel Cell')
			return ?${en_batt}
				[0,	${batt_power_discharge_max_kwac}];
	else if ((technology() == 'PVWatts Battery' ))
		{
			return ?${batt_simple_enable}
				[0,	${batt_simple_kw}];
		}
	else
		return 0;
};
equations { 'fuelcell_power_total' } = define () {
	if( technology() == 'Fuel Cell'){
		return ${fuelcell_power_nameplate};
	}
	else
		return 0;
};

equations { 'fuelcell_total' } = define () {
	return ${fuelcell_power_total}*${fuelcell_per_kW};};

equations { 'system_use_recapitalization' } = define () {
	return 0;
};
