	ssc_var(obj, "tshours", ${tshours});			    //[hrs]
	ssc_var(obj, "T_htf_hot_des", ${T_htf_hot_des});	//[C]
	ssc_var(obj, "T_htf_cold_des", ${T_htf_cold_des});	//[C]
	ssc_var(obj, "rec_htf", ${rec_htf});		        //[-]
	ssc_var(obj, "field_fl_props", ${field_fl_props});	//[-]
	ssc_var(obj, "h_tank_min", ${h_tank_min});		    //[m]
	ssc_var(obj, "h_tank", ${h_tank});				    //[m]
	ssc_var(obj, "tank_pairs", ${tank_pairs});		    //[-]
	ssc_var(obj, "u_tank", ${u_tank});				    //[-]

	ret = ssc_exec(obj, 'ui_tes_calcs');
	
	if(ret==0)
	{
		${Q_tes} = ssc_var(obj, "q_tes");		//[MWt-hr]
		${tes_avail_vol} = ssc_var(obj, "tes_avail_vol");	//[m^3]
		${vol_tank} = ssc_var(obj, "vol_tank");	//[m^3]
		${csp.pt.tes.tank_diameter} = ssc_var(obj, "csp_pt_tes_tank_diameter");			//[m]
		${q_dot_tes_est} = ssc_var(obj, "q_dot_tes_est");	//[MWt]
		${csp.pt.tes.htf_density} = ssc_var(obj, "csp_pt_tes_htf_density");			//[kg/m^3]
	}
	else
	{
		${Q_tes} = 1.0/0.0;
		${tes_avail_vol} = 1.0/0.0;
		${vol_tank} = 1.0/0.0;
		${csp.pt.tes.tank_diameter} = 1.0/0.0;
		${q_dot_tes_est} = 1.0/0.0;
		${csp.pt.tes.htf_density} = 1.0/0.0;
	}
	
	ssc_free( obj );	
};
