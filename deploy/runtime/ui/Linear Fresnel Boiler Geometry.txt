equations{ 'csp.lf.geom1.coll_opt_loss_norm_inc' } = define() {
	return ${csp.lf.geom1.track_error}
*${csp.lf.geom1.geom_error}
*${csp.lf.geom1.mirror_refl}
*${csp.lf.geom1.soiling}
*${csp.lf.geom1.general_error}; };

equations{ 'csp.lf.geom1.avg_field_temp_dt_design' } = define() {
	return (${T_cold_ref}+${T_hot})/2-${T_amb_des_sf}; };

equations{ 'csp.lf.geom1.rec_optical_derate' } = define() {
	return ?${csp.lf.geom1.hl_mode}[ 
	1
    , 
	 ${csp.lf.geom1.var1.field_fraction}*${csp.lf.geom1.var1.bellows_shadowing}*${csp.lf.geom1.var1.hce_dirt}
	+${csp.lf.geom1.var2.field_fraction}*${csp.lf.geom1.var2.bellows_shadowing}*${csp.lf.geom1.var2.hce_dirt}
	+${csp.lf.geom1.var3.field_fraction}*${csp.lf.geom1.var3.bellows_shadowing}*${csp.lf.geom1.var3.hce_dirt}
	+${csp.lf.geom1.var4.field_fraction}*${csp.lf.geom1.var4.bellows_shadowing}*${csp.lf.geom1.var4.hce_dirt} ]; };

equations{ 'csp.lf.geom1.rec_thermal_derate' } = define() {
	return  1 - ${csp.lf.geom1.heat_loss_at_design}/( ${I_bn_des}*${csp.lf.geom1.refl_aper_area}/${csp.lf.geom1.coll_length}); };

equations{ 'csp.lf.geom1.heat_loss_at_design' } = define() {
	return ?${csp.lf.geom1.hl_mode} [ 
	 ${csp.lf.geom1.hlpolyt0}
	+${csp.lf.geom1.hlpolyt1}*${csp.lf.geom1.avg_field_temp_dt_design}
	+${csp.lf.geom1.hlpolyt2}*pow(${csp.lf.geom1.avg_field_temp_dt_design},2)
	+${csp.lf.geom1.hlpolyt3}*pow(${csp.lf.geom1.avg_field_temp_dt_design},3)
        +${csp.lf.geom1.hlpolyt4}*pow(${csp.lf.geom1.avg_field_temp_dt_design},4)
, 
 	 ${csp.lf.geom1.var1.field_fraction} * ${csp.lf.geom1.var1.rated_heat_loss}
	+${csp.lf.geom1.var2.field_fraction} * ${csp.lf.geom1.var2.rated_heat_loss}
	+${csp.lf.geom1.var3.field_fraction} * ${csp.lf.geom1.var3.rated_heat_loss}
	+${csp.lf.geom1.var4.field_fraction} * ${csp.lf.geom1.var4.rated_heat_loss} ]; };

equations{ 'csp.lf.geom1.annulus_gas' } = define()
{
	hvar[0] = ${csp.lf.geom1.var1.gas_type};
	hvar[1] = ${csp.lf.geom1.var2.gas_type};
	hvar[2] = ${csp.lf.geom1.var3.gas_type};
	hvar[3] = ${csp.lf.geom1.var4.gas_type};

	// Case = 1: 'Argon' = 26
	// Case = 2: 'Hydrogen' = 27
	// Case = 3: 'Air' = 1
	for( i = 0; i < 4; i++)
	{
		if(hvar[i] == 1)
			hvar[i] = 26;
		else if(hvar[i] == 2)
			hvar[i] = 27;
		else
			hvar[i] = 1;
	}
	return hvar;
};

equations{ 'csp.lf.geom1.glazing_intact' } = define()
{
	hvar[0] = 1 - ${csp.lf.geom1.var1.broken_glass};
	hvar[1] = 1 - ${csp.lf.geom1.var2.broken_glass};
	hvar[2] = 1 - ${csp.lf.geom1.var3.broken_glass};
	hvar[3] = 1 - ${csp.lf.geom1.var4.broken_glass};

	return hvar;
};
