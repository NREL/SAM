equations{ 'direct_subtotal'} = define() {
  return ${csp.dtr.cost.site_improvements} +
		 ${csp.dtr.cost.solar_field} +
		 ${csp.dtr.cost.htf_system} +
		 ${csp.dtr.cost.storage} +
		 ${csp.dtr.cost.fossil_backup} +
		 ${csp.dtr.cost.power_plant} +
		 ${csp.dtr.cost.bop};
};

equations{ 'total_installed_cost' } = define() {
	return ${total_direct_cost}+${total_indirect_cost}+${csp.dtr.cost.sales_tax.total};
};

equations{ 'csp.dtr.cost.installed_per_capacity' } = define() {
	return 0.001*${total_installed_cost}/${csp.dtr.pwrb.nameplate}; };

equations{ 'csp.dtr.cost.fossil_backup' } = define() {
	return ${csp.dtr.cost.fossil_backup.mwe}*${csp.dtr.cost.fossil_backup.cost_per_kwe}*1000; };

equations{ 'total_indirect_cost' } = define() {
	return ${csp.dtr.cost.epc.total}+${csp.dtr.cost.plm.total}; };

equations{ 'csp.dtr.cost.power_plant.mwe' } = define() {
	return ${P_ref}; };

equations{ 'csp.dtr.cost.fossil_backup.mwe' } = define() {
	return ${P_ref}; };

equations{ 'csp.dtr.cost.nameplate' } = define() {
	return ${csp.dtr.pwrb.nameplate}; };

equations{ 'total_direct_cost' } = define() {
	return ${csp.dtr.cost.contingency}
+${csp.dtr.cost.site_improvements}
+${csp.dtr.cost.solar_field}
+${csp.dtr.cost.htf_system}
+${csp.dtr.cost.storage}
+${csp.dtr.cost.fossil_backup}
+${csp.dtr.cost.power_plant} 
+${csp.dtr.cost.bop}; };

equations{ 'csp.dtr.cost.contingency' } = define() {
	return ${csp.dtr.cost.contingency_percent}/100*(
	 ${csp.dtr.cost.site_improvements}
	+${csp.dtr.cost.solar_field}
	+${csp.dtr.cost.htf_system}
	+${csp.dtr.cost.storage}
	+${csp.dtr.cost.fossil_backup}
	+${csp.dtr.cost.power_plant}
	+${csp.dtr.cost.bop} ); };

equations{ 'csp.dtr.cost.bop' } = define() {
	return ${csp.dtr.cost.bop_mwe}*${csp.dtr.cost.bop_per_kwe}*1000; };

equations{ 'csp.dtr.cost.power_plant' } = define() {
	return ${csp.dtr.cost.power_plant.mwe}*${csp.dtr.cost.power_plant.cost_per_kwe}*1000; };

equations{ 'csp.dtr.cost.sales_tax.value' } = define() {
	return ${sales_tax_rate}; };

equations{ 'csp.dtr.cost.htf_system.area' } = define() {
	return ${total_aperture}; };

equations{ 'csp.dtr.cost.total_land_area' } = define() {
	return ${total_land_area}; };

equations{ 'csp.dtr.cost.solar_field.area' } = define() {
	return ${total_aperture}; };

equations{ 'csp.dtr.cost.plm.total' } = define() {
	return ${csp.dtr.cost.plm.per_acre}*${csp.dtr.cost.total_land_area} 
+ ${csp.dtr.cost.plm.percent}/100*${total_direct_cost} 
+ ${csp.dtr.cost.nameplate}*1000000*${csp.dtr.cost.plm.per_watt} 
+ ${csp.dtr.cost.plm.fixed}; };

equations{ 'csp.dtr.cost.epc.total' } = define() {
	return ${csp.dtr.cost.epc.per_acre}*${csp.dtr.cost.total_land_area} 
+ ${csp.dtr.cost.epc.percent}/100*${total_direct_cost} 
+ ${csp.dtr.cost.nameplate}*1000000*${csp.dtr.cost.epc.per_watt} 
+ ${csp.dtr.cost.epc.fixed}; };

equations{ 'csp.dtr.cost.storage.mwht' } = define() {
	return ${csp.dtr.tes.thermal_capacity}; };
