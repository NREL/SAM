on_change{ 'sca_library' } = define() {

  state = value('use_sca_library_values');
    
  if ( state == 1 ) {  

	value( 'ScaLen'              , value('lib_ScaLen')               );
	value( 'SCA_aper'            , value('lib_SCA_aper')             );
	value( 'ui_sca_aperture_area', value('lib_ui_sca_aperture_area') );
	value( 'Ave_Focal_Length'    , value('lib_Ave_Focal_Length')     );
	value( 'IamF0'               , value('lib_IamF0')                );
	value( 'IamF1'               , value('lib_IamF1')                );
	value( 'IamF2'               , value('lib_IamF2')                );
	value( 'TrkTwstErr'          , value('lib_TrkTwstErr')           );
	value( 'GeoAcc'              , value('lib_GeoAcc')               );
	value( 'MirRef'              , value('lib_MirRef')               );
	value( 'MirCln'              , value('lib_MirCln')               );
	value( 'ui_HCEdust'          , value('lib_HCEdust')              );
	value( 'ConcFac'             , value('lib_ConcFac')              );

  }
