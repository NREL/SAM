function enable_revenue_escalation_rate() 
{
	enable('subscriber1_payment_annual_escal', #${subscriber1_payment_annual} == 1 ); 
	enable('subscriber2_payment_annual_escal', #${subscriber2_payment_annual} == 1 ); 
	enable('subscriber3_payment_annual_escal', #${subscriber3_payment_annual} == 1 ); 
	enable('subscriber4_payment_annual_escal', #${subscriber4_payment_annual} == 1 ); 
	enable('subscriber1_payment_generation_escal', #${subscriber1_payment_generation} == 1 ); 
	enable('subscriber2_payment_generation_escal', #${subscriber2_payment_generation} == 1 ); 
	enable('subscriber3_payment_generation_escal', #${subscriber3_payment_generation} == 1 ); 
	enable('subscriber4_payment_generation_escal', #${subscriber4_payment_generation} == 1 ); 
	enable('unsubscribed_payment_generation_escal', #${unsubscribed_payment_generation} == 1 ); 
}

function duplicate_revenue_warning()
{
	str = '';
	msg = '';
	if ( sum(${subscriber1_payment_annual}) > 0 && sum(${subscriber1_payment_generation}) > 0 )
	{ 
		str += ? (${ui_subscriber1_name}=='') [${ui_subscriber1_name}, '1' ]; 
	}
	if ( sum(${subscriber2_payment_annual}) > 0 && sum(${subscriber2_payment_generation}) > 0 )
	{ 
		if ( str != '' ) 
		{ 
			str += ', ';
		} 
		str += ? (${ui_subscriber2_name}=='') [${ui_subscriber2_name}, '2' ]; 
	}
	if ( sum(${subscriber3_payment_annual}) > 0 && sum(${subscriber3_payment_generation}) > 0 )
	{ 
		if ( str != '' ) 
		{ 
			str += ', ';
		} 
		str += ? (${ui_subscriber3_name}=='') [${ui_subscriber3_name}, '3' ]; 
	}
	if ( sum(${subscriber4_payment_annual}) > 0 && sum(${subscriber4_payment_generation}) > 0 )
	{ 
		if ( str != '' ) 
		{ 
			str += ', ';
		} 
		str += ? (${ui_subscriber4_name}=='') [${ui_subscriber4_name}, '4' ]; 
	}
	if ( str != '' ) 
	{ 
		msg = 'The following subscriber classes have revenue from both annual ($/yr) and generation ($/kWh) payments: ' + str + '. Community solar projects typically earn revenue from one or the other.'; 
	}
	property( 'lbl_revenue_message','Caption',msg);
	property( 'lbl_revenue_message','TextColour','red');

}
