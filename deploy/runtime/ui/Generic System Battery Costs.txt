equations{ 'nameplate_capacity' } = define() {
	return ${system_capacity}; };

equations{ 'battery_energy' } = define() {
	return ${batt_computed_bank_capacity}; };

equations{ 'battery_power' } = define() {
	return ${batt_power_discharge_max_kwdc};};

equations{ 'battery_total' } = define() {
	return ${battery_energy}*${battery_per_kWh} + ${battery_power}*${battery_per_kW} ;
};

equations{ 'total_installed_cost' } = define() {
	return ${total_direct_cost}+${total_indirect_cost}+${genericsys.cost.sales_tax.total}; };

equations{ 'genericsys.cost.installed_per_capacity' } = define() {
	return ${total_installed_cost}/${system_capacity}/1000; };

equations{ 'genericsys.cost.epc.nonfixed' } = define() {
	return ${total_direct_cost}*${genericsys.cost.epc.percent}/100; };

equations{ 'genericsys.cost.sales_tax.value' } = define() {
	return ${sales_tax_rate}; };

equations{ 'genericsys.cost.contingency' } = define() {
	return ${genericsys.cost.contingency_percent}/100
*(${genericsys.cost.plant}+${genericsys.cost.plant_scaled}+${battery_total}); };

equations{ 'total_indirect_cost' } = define() {
	return ${genericsys.cost.epc.total}+${genericsys.cost.plm.total}; };

equations{ 'genericsys.cost.sales_tax.total' } = define() {
	return ${genericsys.cost.sales_tax.value}/100*${total_direct_cost}*${genericsys.cost.sales_tax.percent}/100; };

equations{ 'total_direct_cost' } = define() {
	return ${genericsys.cost.contingency}
+${genericsys.cost.plant}
+${genericsys.cost.plant_scaled} +${battery_total}; };

equations{ 'genericsys.cost.plant'} = define() {
	return ${fixed_plant_input};
};
equations{ 'genericsys.cost.plm.nonfixed' } = define() {
	return ${total_direct_cost}*${genericsys.cost.plm.percent}/100; };

equations{ 'genericsys.cost.epc.total' } = define() {
	return ${genericsys.cost.epc.fixed}+${genericsys.cost.epc.nonfixed}; };

equations{ 'genericsys.cost.plm.total' } = define() {
	return ${genericsys.cost.plm.fixed}+${genericsys.cost.plm.nonfixed}; };

equations{ 'genericsys.cost.plant_scaled' } = define() {
	return ${system_capacity}*1000*${genericsys.cost.per_watt};
};

equations { 'system_use_recapitalization' } = define () {
		return 0;
