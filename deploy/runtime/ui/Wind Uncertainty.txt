on_change{'correlation_uncert'} = define() {
	mat = ${correlation_uncert};
	// make sure matrix is symmetrical with ones on diagonal
	for ( i = 0; i < #mat; i++) {
		for ( j = 0; j < i; j++) {
			val = mat[j][i];
			if (val <= 1 && val >= -1)
				mat[i][j] = val;
			else if (val > 1){
				mat[i][j] = 1;
				mat[j][i] = 1;
			}
			else{
				mat[i][j] = -1;
				mat[j][i] = -1;
			}
		}
		mat[i][i] = 1;
	}
	value('correlation_uncert', mat, false);
	refresh('correlation_uncert');
	calculate_uncertainties();
};

function combine_uncertainty(year) {
	eval_period = ${eval_period_uncert};
	eval_perf = ${eval_perf_uncert};
	eval_climate = ${eval_climate_uncert};
	if (year == 10){
		eval_period = ${eval_period_10yr_uncert};
		eval_perf = ${eval_perf_10yr_uncert};
		eval_climate = ${eval_climate_10yr_uncert};
	}
	else if (year == 20){
		eval_period = ${eval_period_20yr_uncert};
		eval_perf = ${eval_perf_20yr_uncert};
		eval_climate = ${eval_climate_20yr_uncert};
	}
	components = [${resource_period_uncert}, ${resource_refdata_uncert},
		${resource_adj_uncert}, ${resource_dist_uncert},
		${resource_datasyn_uncert}, eval_period,
		eval_climate, eval_perf, ${meas_speed_uncert},
		${meas_dir_uncert}, ${meas_atm_uncert}, ${meas_data_uncert},
		${hextrap_inputs_uncert}, ${hextrap_sens_uncert}, ${hextrap_approp_uncert},
		${vextrap_inputs_uncert}, ${vextrap_comp_uncert}, ${vextrap_stress_uncert},
		${wake_int_uncert}, ${wake_ext_uncert}, ${wake_future_uncert},
		${avail_turb_uncert}, ${avail_bop_uncert}, ${avail_grid_uncert},
		${elec_eff_uncert}, ${elec_parasitic_uncert}, ${turb_subopt_uncert},
		${turb_generic_uncert}, ${turb_specific_uncert}, ${turb_hysteresis_uncert},
		${env_icing_uncert}, ${env_degrad_uncert}, ${env_env_uncert}, ${env_exp_uncert},
		${ops_load_uncert}, ${ops_grid_uncert}, ${ops_env_uncert}, ${ops_strategies_uncert}];
	n_wspd_comps = 18;
	sens = ${sensitivity_uncert};
	correlation = ${correlation_uncert};
	for ( i = 0; i < n_wspd_comps; i++ ){
		components[i] = sens*components[i];
	}
	for ( i = 0; i < #components; i++ ){
		components[i] /= 100.;
	}
	total = 0;
	wspd = 0;
	for ( i = 0; i < #components; i++ ){
		comp = components[i];
		total += pow(comp, 2);
		for ( j = 0; j < i; j++ ){
			total += 2*comp*components[j]*correlation[i][j];
		}
		if (i == n_wspd_comps - 1){
			wspd = sqrt(total) * 100.;
		}
	}
	if (year == 10)
		value('total_10yr_uncert', sqrt(total)*100.);
	else if (year == 20)
		value('total_20yr_uncert', sqrt(total)*100.);
	else
		value('total_uncert', sqrt(total)*100.);
	value('wspd_uncert', wspd);
}
