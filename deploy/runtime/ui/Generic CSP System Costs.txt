equations{ 'csp.gss.cost.plm.total' } = define() {
	return ${csp.gss.cost.plm.per_acre}*${csp.gss.cost.total_land_area} 
+ ${csp.gss.cost.plm.percent}/100*${total_direct_cost} 
+ ${csp.gss.cost.nameplate}*1000000*${csp.gss.cost.plm.per_watt} 
+ ${csp.gss.cost.plm.fixed}; };

equations{ 'csp.gss.cost.sales_tax.value' } = define() {
	return ${sales_tax_rate}; };

equations{ 'csp.gss.cost.bop_mwe' } = define() {
	return ${w_des}; };

equations{ 'total_installed_cost' } = define() {
	return ${total_direct_cost}+${total_indirect_cost}+${csp.gss.cost.sales_tax.total}; };

equations{ 'csp.gss.cost.solar_field.area' } = define() {
	return ${csp.gss.sf.field_area}; };

equations{ 'csp.gss.cost.contingency' } = define() {
	return ${csp.gss.cost.contingency_percent}/100*(
	${csp.gss.cost.solar_field}
	+${csp.gss.cost.storage}
	+${csp.gss.cost.power_plant} 
	+${csp.gss.cost.site_improvements}
	+${csp.gss.cost.fossil_backup}
	+${csp.gss.cost.bop}); };

equations{ 'total_indirect_cost' } = define() {
	return ${csp.gss.cost.epc.total}+${csp.gss.cost.plm.total}; };

equations{ 'csp.gss.cost.sales_tax.total' } = define() {
	return ${csp.gss.cost.sales_tax.value}/100*${total_direct_cost}*${csp.gss.cost.sales_tax.percent}/100; };

equations{ 'csp.gss.cost.power_plant' } = define() {
	return ${csp.gss.cost.power_plant.mwe}*${csp.gss.cost.power_plant.cost_per_kwe}*1000; };

equations{ 'csp.gss.cost.fossil_backup.mwe' } = define() {
	return ${w_des}; };

equations{ 'csp.gss.cost.site_improvements' } = define() {
	return ${csp.gss.cost.site_improvements.area}*${csp.gss.cost.site_improvements.cost_per_m2}; };

equations{ 'total_direct_cost' } = define() {
	return ${csp.gss.cost.contingency}
+${csp.gss.cost.solar_field}
+${csp.gss.cost.storage}
+${csp.gss.cost.power_plant}
+${csp.gss.cost.site_improvements}
+${csp.gss.cost.fossil_backup}
+${csp.gss.cost.bop}; };

equations{ 'csp.gss.cost.nameplate' } = define() {
	return ${csp.gss.pwrb.nameplate}; };

equations{ 'csp.gss.cost.total_land_area' } = define() {
	return ${csp.gss.solf.total_land_area}; };

equations{ 'csp.gss.cost.fossil_backup' } = define() {
	return ${csp.gss.cost.fossil_backup.mwe}*${csp.gss.cost.fossil_backup.cost_per_kwe}*1000; };

equations{ 'csp.gss.cost.epc.total' } = define() {
	return ${csp.gss.cost.epc.per_acre}*${csp.gss.cost.total_land_area} 
+ ${csp.gss.cost.epc.percent}/100*${total_direct_cost} 
+ ${csp.gss.cost.nameplate}*1000000*${csp.gss.cost.epc.per_watt} 
+ ${csp.gss.cost.epc.fixed}; };

equations{ 'csp.gss.cost.storage.mwht' } = define() {
	return ${csp.gss.tes.max_capacity}; };

equations{ 'csp.gss.cost.storage' } = define() {
	return ${csp.gss.cost.storage.mwht}*${csp.gss.cost.storage.cost_per_kwht}*1000; };

equations{ 'csp.gss.cost.site_improvements.area' } = define() {
	return ${csp.gss.sf.field_area}; };

equations{ 'csp.gss.cost.bop' } = define() {
	return ${csp.gss.cost.bop_mwe}*${csp.gss.cost.bop_per_kwe}*1000; };

	return ${csp.gss.cost.solar_field.area}*${csp.gss.cost.solar_field.cost_per_m2}; };

equations{ 'csp.gss.cost.power_plant.mwe' } = define() {
	return ${w_des}; };
