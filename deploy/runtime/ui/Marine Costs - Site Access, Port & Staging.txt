equations {'staging_port_cost_total'} = define() {
	return ${quay_cost}+${laydown_area_cost}+
		${assembly_areas_cost}+${personnel_facilities_cost}+
		${fabrication_facilities_cost}+${other_staging_port_cost};
};

equations {'staging_port_cost_total_check'} = define() {
	if (${detailed_cost_choice_staging_port}==0)
	{
		return ${staging_port_cost};
	}
	else
	{
		return ${staging_port_cost_total};
	}
};

equations {'port_fees_cost_total'} = define() {
	return ${entrance_exit_fees_cost}+${quayside_docking_fees_cost}+
		${wharfage_fees_cost}+${other_port_fees_cost};
};

equations {'port_fees_cost_total_check'} = define() {
	if (${detailed_cost_choice_port_fees}==0)
	{
		return ${port_fees_cost};
	}
	else
	{
		return ${port_fees_cost_total};
	}
};

equations {'site_access_port_staging_cost_total'} = define() {
	return ${staging_port_cost_total_check}+${cranage_cost}+${port_improvement_cost}+
		${port_fees_cost_total_check}+${other_site_access_cost};
};

equations {'site_access_port_staging_cost'} = define() {
	return ?${site_access_port_staging_cost_method}
			[
				${site_access_port_staging_cost_input}*${system_capacity},
				${site_access_port_staging_cost_input},
				${site_access_port_staging_cost_modeled_ui},
				${site_access_port_staging_cost_total}
			];
};

equations {'itemized_check_site_access'} = define() {
	if (${balance_of_system_cost_override}!=1)
	{
		return ?${site_access_port_staging_cost_method}
				[
					0,
					0,
					2,
					1
				];
	}
	else
	{
		return 2;
	}
};

equations {'site_access_port_staging_cost_modeled_ui'} = define() {
	return 0.011*${capex_check};
};

on_load{'Marine Costs - Site Access, Port & Staging'} = define() {
	on_change{'itemized_check_site_access'};
	enable_itemized_site_access();
	on_change{'detailed_cost_choice_port_fees'};
	detailed_cost_port_fees();
	on_change{'detailed_cost_choice_staging_port'};
	detailed_cost_staging_port();
	//enable_itemized_level();
	check_override_site_access();
};

function check_override_site_access()
{
	check_value = value('balance_of_system_cost_override');
	enable('site_access_port_staging_cost_method',check_value!=1);
	enable('site_access_port_staging_cost',check_value!=1);
	show('site_override_label',check_value==1);
	
	refresh();
}	

on_change{'detailed_cost_choice_staging_port'} = define() {
	detailed_cost_staging_port();
};

function detailed_cost_staging_port()
{
	if (value('itemized_check_site_access')==1)
	{
		detailed_value = value('detailed_cost_choice_staging_port');
		enable('staging_port_cost',detailed_value==0);
		show('staging_port_cost',detailed_value==0);
		enable('staging_port_cost_total',detailed_value);
		show('staging_port_cost_total',detailed_value);
		enable('quay_cost',detailed_value);
		enable('laydown_area_cost',detailed_value);
		enable('assembly_areas_cost',detailed_value);
		enable('personnel_facilities_cost',detailed_value);
		enable('fabrication_facilities_cost',detailed_value);
		enable('other_staging_port_cost',detailed_value);
		
		refresh();
	}
}

on_change{'detailed_cost_choice_port_fees'} = define() {
	detailed_cost_port_fees();
};

function detailed_cost_port_fees()
{
	if (value('itemized_check_site_access')==1)
	{
		detailed_value = value('detailed_cost_choice_port_fees');
		enable('port_fees_cost',detailed_value==0);
		show('port_fees_cost',detailed_value==0);
		enable('port_fees_cost_total',detailed_value);
		show('port_fees_cost_total',detailed_value);
		enable('entrance_exit_fees_cost',detailed_value);
		enable('quayside_docking_fees_cost',detailed_value);
		enable('wharfage_fees_cost',detailed_value);
		enable('other_port_fees_cost',detailed_value);
		
		refresh();
	}
}

on_change{'itemized_check_site_access'} = define() {
	enable_itemized_site_access();
	
};

function enable_itemized_site_access()
{
	itemized_value = value('itemized_check_site_access');
	enable('staging_port_cost',itemized_value==1);
	enable('staging_port_cost_total',itemized_value==1);
	enable('detailed_cost_choice_staging_port',itemized_value==1);
	enable('quay_cost',itemized_value==1);
	enable('laydown_area_cost',itemized_value==1);
	enable('assembly_areas_cost',itemized_value==1);
	enable('personnel_facilities_cost',itemized_value==1);
	enable('fabrication_facilities_cost',itemized_value==1);
	enable('other_staging_port_cost',itemized_value==1);
	enable('cranage_cost',itemized_value==1);
	enable('port_improvement_cost',itemized_value==1);
	enable('port_fees_cost',itemized_value==1);
	enable('port_fees_cost_total',itemized_value==1);
	enable('detailed_cost_choice_port_fees',itemized_value==1);
	enable('entrance_exit_fees_cost',itemized_value==1);
	enable('quayside_docking_fees_cost',itemized_value==1);
	enable('wharfage_fees_cost',itemized_value==1);
	enable('other_port_fees_cost',itemized_value==1);
	enable('other_site_access_cost',itemized_value==1);
	enable('site_access_port_staging_cost_input',itemized_value!=2);
	show('site_access_port_staging_cost_total',itemized_value==1);
	show('site_access_port_staging_cost_total_label',itemized_value==1);
	show('site_access_port_staging_cost_input',(itemized_value==0 || itemized_value==2));
	show('site_access_port_staging_cost_input_label',(itemized_value==0 || itemized_value==2));
	if (itemized_value==1)
	{
		detailed_cost_port_fees();
		detailed_cost_staging_port();
	}
	/*if (value('itemized_check_eng')==1)
	{
		enable_itemized_level_eng();
	}*/
	refresh();
}
