	if (${csp.pt.pwrb.pressure_mode} == 1) //sliding pressure
		return 3; // user defined
		else
		if(${csp.pt.pwrb.pressure_mode}==2) //first custom cycle
		return 5;
		else
		if(${csp.pt.pwrb.pressure_mode}==3) //second custom cycle
		return 6;
	else // fixed pressure
		return 1; // tower
