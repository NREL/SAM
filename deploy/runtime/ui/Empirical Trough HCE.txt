  return ${calc_col_factor};

// Optical Efficiency (HCE) for Receivers 1 through 4
equations{ 'ui_hce_opt_eff_1' } = define() {
	return ${calc_hce_col_factor}
    * ?${ui_hce_broken_glass_1}[${ui_hce_HCEdust},1.0]
    * ${HCEBelShad_1}
    * ${HCEEnvTrans_1}
    * ${HCEabs_1}
    * ${HCEmisc_1}; 
};

equations{ 'ui_hce_opt_eff_2' } = define() {
	return ${calc_hce_col_factor}
    *? ${ui_hce_broken_glass_2}[${ui_hce_HCEdust},1.0]
    * ${HCEBelShad_2}
    * ${HCEEnvTrans_2}
    * ${HCEabs_2}
    * ${HCEmisc_2}; 
};

equations{ 'ui_hce_opt_eff_3' } = define() {
	return ${calc_hce_col_factor}
    * ?${ui_hce_broken_glass_3}[${ui_hce_HCEdust},1.0]
    * ${HCEBelShad_3}
    * ${HCEEnvTrans_3}
    * ${HCEabs_3}
    * ${HCEmisc_3}; 
};

equations{ 'ui_hce_opt_eff_4' } = define() {
	return ${calc_hce_col_factor}
    * ?${ui_hce_broken_glass_4}[${ui_hce_HCEdust},1.0]
    * ${HCEBelShad_4}
    * ${HCEEnvTrans_4}
    * ${HCEabs_4}
    * ${HCEmisc_4}; 
};

// Optical Efficiency (Weighted)
equations{ 'ui_hce_opt_eff_weighted' } = define() {
	return  ${ui_hce_opt_eff_1}*${HCEFrac_1} 
	+ ${ui_hce_opt_eff_2}*${HCEFrac_2} 
	+ ${ui_hce_opt_eff_3}*${HCEFrac_3} 
	+ ${ui_hce_opt_eff_4}*${HCEFrac_4}; 
};

// Internal coefficients for heat loss calculations
equations{ 'ui_hce_hl_term_1' } = define() {
	return ${ui_reference_wind_speed}^0.5; 
};
	
equations{ 'ui_hce_hl_term_2' } = define() {
	return 0.5*(  ${SfOutTempD}
     + ${SfInTempD} )
     - ${ui_reference_ambient_temperature}; 
};

equations{ 'ui_hce_hl_term_3' } = define() {
	return ( ${SfOutTempD}^2
     + ${SfOutTempD} * ${SfInTempD}
     + ${SfInTempD}^2 ) / 3.0; 
};

equations{ 'ui_hce_hl_term_4' } = define() {
	return ( ${SfOutTempD}^2 + ${SfInTempD}^2 ) 
	 * ( ${SfOutTempD} + ${SfInTempD} ) / 4.0; 
};

// HCE Heat Losses (W/m) for Receivers 1 through 4
equations{ 'ui_hce_heat_losses_1' } = define() {
	return ${PerfFac_1} * ( 
	 ( ${HCEA0_1} + ${HCEA5_1} * ${ui_hce_hl_term_1} )
     + (${HCEA1_1}
     + ${HCEA6_1} * ${ui_hce_hl_term_1}) * ${ui_hce_hl_term_2}
     + ( ${HCEA2_1} + ${HCEA4_1} * ${ui_reference_direct_normal_irradiance} ) * ${ui_hce_hl_term_3}
     + ${HCEA3_1} * ${ui_hce_hl_term_4} ); 
};
 
 equations{ 'ui_hce_heat_losses_2' } = define() {
	return ${PerfFac_2} * (
	 ( ${HCEA0_2} + ${HCEA5_2} * ${ui_hce_hl_term_1} )
     + ( ${HCEA1_2} + ${HCEA6_2} * ${ui_hce_hl_term_1} ) * ${ui_hce_hl_term_2}
     + ( ${HCEA2_2} + ${HCEA4_2} * ${ui_reference_direct_normal_irradiance} ) * ${ui_hce_hl_term_3}
     + ${HCEA3_2} * ${ui_hce_hl_term_4} ); 
};

equations{ 'ui_hce_heat_losses_3' } = define() {
	return ${PerfFac_3} * ( 
	 ( ${HCEA0_3} + ${HCEA5_3} * ${ui_hce_hl_term_1} )
	 + ( ${HCEA1_3} + ${HCEA6_3} * ${ui_hce_hl_term_1} ) * ${ui_hce_hl_term_2}
	 + ( ${HCEA2_3} + ${HCEA4_3} * ${ui_reference_direct_normal_irradiance}) * ${ui_hce_hl_term_3}
	 + ${HCEA3_3} * ${ui_hce_hl_term_4} ); 
};

equations{ 'ui_hce_heat_losses_4' } = define() {
	return ${PerfFac_4} * (
	 ( ${HCEA0_4} + ${HCEA5_4} * ${ui_hce_hl_term_1} )
	 + ( ${HCEA1_4} + ${HCEA6_4} * ${ui_hce_hl_term_1}) * ${ui_hce_hl_term_2}
	 + ( ${HCEA2_4} + ${HCEA4_4} * ${ui_reference_direct_normal_irradiance}) * ${ui_hce_hl_term_3}
	 + ${HCEA3_4} * ${ui_hce_hl_term_4} ); 
};

// Thermal Losses (Weighted W/m)
equations{ 'ui_hce_thermloss_weighted_m' } = define() {
	return ${ui_hce_heat_losses_1} * ${HCEFrac_1}
	 + ${ui_hce_heat_losses_2} * ${HCEFrac_2}
	 + ${ui_hce_heat_losses_3} * ${HCEFrac_3}
	 + ${ui_hce_heat_losses_4} * ${HCEFrac_4}; 
};

// Thermal Losses (Weighted W/m2)
equations{ 'ui_hce_thermloss_weighted_m2' } = define() {
	return ${ui_hce_thermloss_weighted_m} / ${SCA_aper}; 
};
