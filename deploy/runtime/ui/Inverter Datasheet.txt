function inverter_datasheet_value_change()
{
	efficiency_type = value('inv_ds_eff_type');

	weighted_efficiency = (efficiency_type == 0);
	
	enable('inv_ds_eff_weighted',weighted_efficiency);
	enable('inv_ds_eff_peak_or_nom',!weighted_efficiency);
	enable('inv_ds_pso',!weighted_efficiency);

}
	
	
on_change{'inv_ds_eff_type'} = define()
{
	efficiency_type = value('inv_ds_eff_type');

	weighted_efficiency = (efficiency_type == 0);
	
	if (weighted_efficiency)
	{
		value('inv_ds_pso',0);
	}

	enable('inv_ds_pso',!weighted_efficiency);
	inverter_datasheet_value_change();
};

	
on_load{"Inverter Datasheet"}=define(){
	inverter_datasheet_value_change();	
};	


on_change{'btnSaveInvDSData'} = define() {
	logmsg('running btnSaveInvDSData');
	file = choose_file( '', 'Save Inverter Datasheet Data As', 
		'Data Files (*.csv)|*.csv', true);
		
	if ( file != '' ) {
		file = replace(file, '\\', '/');
		if (group_write('PV Inverter (Datasheet)',file))
		{	
			value( 'inv_ds_data_filename', file );
			msgbox("Data saved! \n");
		}
		else
		{
			value( 'inv_ds_data_filename', "" );
			msgbox("Data save failed! \n");
		}
	}
};

on_change{'btnLoadInvDSData'} = define() {
	logmsg('running btnLoadInvDSData');
	file = choose_file( '', 'Open Inverter Datasheet File', 
		'Data Files (*.csv)|*.csv');

		else
		{
			value( 'inv_ds_data_filename', "" );
			msgbox("Data load failed! \n");
		}
	}
};
