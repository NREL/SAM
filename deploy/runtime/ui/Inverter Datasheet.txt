function inverter_datasheet_value_change()
{
	efficiency_type = value('inv_ds_eff_type');

	weighted_efficiency = (efficiency_type == 0);
	
	enable('inv_ds_eff_weighted',weighted_efficiency);
	enable('inv_ds_eff_peak_or_nom',!weighted_efficiency);
	enable('inv_ds_pso',!weighted_efficiency);

}

on_change{'btn_copy_from_library'} = define()
{

    value('inv_ds_paco', value('inv_snl_paco'));
    value('inv_ds_pnt', value('inv_snl_pnt'));
    value('inv_ds_pso', value('inv_snl_pso'));
    value('inv_ds_vdco', value('inv_snl_vdco'));
    value('inv_ds_vdcmax', value('inv_snl_vdcmax'));
    
    value('inv_ds_eff_type', 1);
    value('inv_ds_eff_peak_or_nom', value('inv_snl_paco') / value('inv_snl_pdco') * 100 );
    value('inv_ds_eff', value('inv_snl_eff_cec'));
    
    value('inv_ds_ac_voltage', value('inv_snl_ac_voltage'));
    value('inv_ds_idcmax', value('inv_snl_idcmax'));
    value('inv_ds_mppt_low', value('inv_snl_mppt_low'));
    value('inv_ds_mppt_hi', value('inv_snl_mppt_hi'));
    value('inv_ds_num_mppt', value('inv_snl_num_mppt'));
    value('inv_ds_eff_weighted', value('inv_snl_eff_cec'));

	msgbox('Inverter data copied.\nCopied data for ' + value('inv_cec_name') + ' from the inverter library.');

};	
	
on_change{'inv_ds_eff_type'} = define()
{
	efficiency_type = value('inv_ds_eff_type');

	weighted_efficiency = (efficiency_type == 0);
	
	if (weighted_efficiency)
	{
		value('inv_ds_pso',0);
	}

	enable('inv_ds_pso',!weighted_efficiency);
	inverter_datasheet_value_change();
};

on_change{'btnSaveInvDSData'} = define() {
	logmsg('running btnSaveInvDSData');
	file = choose_file( '', 'Save Inverter Datasheet Data As', 
		'Data Files (*.csv)|*.csv', true);
		
	if ( file != '' ) {
		file = replace(file, '\\', '/');
		if (group_write('PV Inverter (Datasheet)',file))
		{	
			value( 'inv_ds_data_filename', file );
			msgbox("Data saved! \n");
		}
		else
		{
			value( 'inv_ds_data_filename', "" );
			msgbox("Data save failed! \n");
		}
	}
};

on_change{'btnLoadInvDSData'} = define() {

		}
		else
		{
			value( 'inv_ds_data_filename', "" );
			msgbox("Data load failed! \n");
		}
	}
};
