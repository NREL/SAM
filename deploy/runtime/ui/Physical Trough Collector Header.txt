equations{ '$MIMO$ collectors in field' } = define () {
  arr = ${SCAInfoArray};
  n = ${nColt};
  str = "Cold - "; 
  for ( i=0; i<#arr; i++ ) // number of collector assemblies in loop
    for ( j=0; j<n; j++ ) 
      if ( arr[i][0] == j+1 ) { // collector type number for UI display (1-4)
        s[j] = to_string(j+1);
        str = str + s[j] + " - ";
	  }
  ${collectors_in_field} = str + "Hot";
};
