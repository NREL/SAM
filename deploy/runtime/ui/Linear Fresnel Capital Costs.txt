equations{ 'actual_aper' } = define() {
	tech = technology();
	aa = 0;
	if (tech == "DSLF")
		aa = ${csp.lf.sf.dp.actual_aper};
	if (tech == "MSLF")
		aa = ${a_sf_act};
	return aa;
};

equations{ 'csp.lf.cost.total_installed' } = define() {
	return ${csp.lf.cost.total_direct}+${csp.lf.cost.total_indirect}; };

equations{ 'csp.lf.cost.bop' } = define() {
	return ${csp.lf.cost.bop_mwe}*${csp.lf.cost.bop_per_kwe}*1000; };

equations{ 'csp.lf.cost.fossil_backup.mwe' } = define() {
	return ${demand_var}; };

equations{ 'csp.lf.cost.total_direct' } = define() {
	return ${csp.lf.cost.contingency}
+${csp.lf.cost.site_improvements}
+${csp.lf.cost.solar_field}
+${csp.lf.cost.htf_system}
+${csp.lf.cost.fossil_backup}
+${csp.lf.cost.power_plant} 
+${csp.lf.cost.bop}; };

equations{ 'csp.lf.cost.sales_tax.total' } = define() {
	return ${csp.lf.cost.sales_tax.value}/100*${csp.lf.cost.total_direct}*${csp.lf.cost.sales_tax.percent}/100; };

equations{ 'csp.lf.cost.total_land_area' } = define() {
	tech = technology();
	la = 0;
	if (tech == "DSLF")
		la = ${csp.lf.sf.total_land_area};
	if (tech == "MSLF")
		la = ${total_land_area};
	return la; };

equations{ 'csp.lf.cost.installed_per_capacity' } = define() {
	return 0.001*${csp.lf.cost.total_installed}/${nameplate}; };

equations{ 'csp.lf.cost.solar_field.area' } = define() {
	return ${actual_aper}; };

equations{ 'csp.lf.cost.sales_tax.value' } = define() {
	return ${sales_tax_rate}; };

equations{ 'csp.lf.cost.fossil_backup' } = define() {
	return ${csp.lf.cost.fossil_backup.mwe}*${csp.lf.cost.fossil_backup.cost_per_kwe}*1000; };

equations{ 'csp.lf.cost.power_plant' } = define() {
	return ${csp.lf.cost.power_plant.mwe}*${csp.lf.cost.power_plant.cost_per_kwe}*1000; };

equations{ 'csp.lf.cost.bop_mwe' } = define() {
	return ${demand_var}; };

equations{ 'csp.lf.cost.site_improvements.area' } = define() {
	return ${actual_aper}; };

equations{ 'csp.lf.cost.total_indirect' } = define() {
	return ${csp.lf.cost.epc.total}+${csp.lf.cost.plm.total}+${csp.lf.cost.sales_tax.total}; };

equations{ 'csp.lf.cost.site_improvements' } = define() {
	return ${csp.lf.cost.site_improvements.area}*${csp.lf.cost.site_improvements.cost_per_m2}; };

equations{ 'csp.lf.cost.nameplate' } = define() {
	return ${nameplate}; };

equations{ 'csp.lf.cost.epc.total' } = define() {
	return ${csp.lf.cost.epc.per_acre}*${csp.lf.cost.total_land_area} 
+ ${csp.lf.cost.epc.percent}/100*${csp.lf.cost.total_direct} 
+ ${csp.lf.cost.nameplate}*1000000*${csp.lf.cost.epc.per_watt} 
+ ${csp.lf.cost.epc.fixed}; };

equations{ 'csp.lf.cost.contingency' } = define() {
	return ${csp.lf.cost.contingency_percent}/100*(
	 ${csp.lf.cost.site_improvements}
	+${csp.lf.cost.solar_field}
	+${csp.lf.cost.htf_system}
	+${csp.lf.cost.fossil_backup}
	+${csp.lf.cost.power_plant}
	+${csp.lf.cost.bop} ); };

equations{ 'csp.lf.cost.htf_system.area' } = define() {
	return ${actual_aper}; };

equations{ 'csp.lf.cost.htf_system' } = define() {
	return ${csp.lf.cost.htf_system.area}*${csp.lf.cost.htf_system.cost_per_m2}; };

equations{ 'csp.lf.cost.solar_field' } = define() {
	return ${csp.lf.cost.solar_field.area}*${csp.lf.cost.solar_field.cost_per_m2}; };

equations{ 'csp.lf.cost.plm.total' } = define() {
	return ${csp.lf.cost.plm.per_acre}*${csp.lf.cost.total_land_area}

+ ${csp.lf.cost.nameplate}*1000000*${csp.lf.cost.plm.per_watt} 
+ ${csp.lf.cost.plm.fixed}; };

	return ${demand_var};
