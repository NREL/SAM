on_change{'detailed_cost_choice_marine_systems'} = define() {
	detailed_cost_marine_systems();
};

function detailed_cost_marine_systems()
{
	if (value('itemized_check_structural')==1)
	{
		detailed_value_marine = value('detailed_cost_choice_marine_systems');
		enable('marine_systems_cost',detailed_value_marine==0);
		show('marine_systems_cost',detailed_value_marine==0);
		show('marine_systems_cost_total',detailed_value_marine);
		enable('marine_systems_cost_total',detailed_value_marine);
		enable('personnel_access_system_cost',detailed_value_marine);
		enable('ballast_system_cost',detailed_value_marine);
		enable('navigation_lighting_cost',detailed_value_marine);
		enable('other_marine_system_cost',detailed_value_marine);
		show('warning_marine_systems',detailed_value_marine==2);
		
		refresh();
	}
}
