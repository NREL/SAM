function interpolate_batt_capex_table()

	input = ${capex_table_batt_inputs};
	
	sys_cap = ${battery_energy_kwh};

	nr = #input;
	nc = #input[0];
	//refresh();
	idx=0;
	//Cap_low = [];
	//Cap_high = [];
	if (nc == 1) {
		cost = 0;
		total_cost = cost*sys_cap;
		value("capex_table_batt_cost_per_kw", cost);
		value("capex_table_total_installed_batt_cost",total_cost);
		property('capex_batt_warning_message', 'Caption', 'The cost table must have at least two columns: Capacity (kWh) and Costs ($/kWdc). A third column can also be used for Costs ($/kWh)');
		exit;
	}
	else if (nc == 2) {
		if (sys_cap > input[nr - 1][0]) {
			cost = input[nr - 1][1];

		}
		else if (sys_cap < input[0][0]) {
			cost = input[0][1];
			cap_string = to_string(to_int(sys_cap));
			table_string = to_string(to_int(input[0][0]));
			property('capex_batt_warning_message','Caption','The DC battery system capacity is '+cap_string+' kWh and the minimum costs table DC battery capacity value is '+table_string + ' kWh. The system capacity is outside the bounds of the costs table.');

			property('capex_batt_warning_message','Caption','');	
		}
		total_cost = cost*sys_cap + cost_kwh * sys_kwh_cap;

		else if (sys_cap < input[0][0]) {

		}
		total_cost = cost*sys_cap + cost_kwh * sys_kwh_cap;
		value("capex_table_batt_cost_per_kw", cost);
