function toggle_dispatch_fc()
{
	en_batt = ${en_batt};
	mode = ${batt_dispatch_choice_ui};
	fc_mode = ${fuelcell_dispatch_choice};
	btm = behind_the_meter();
	allow_grid_charge = en_batt && (mode == 3);
	allow_pv_charge = en_batt && (mode == 3);
	allow_discharge = en_batt && (mode == 3);
	allow_fc_charge = fc_mode == 2;
	allow_fc_discharge = fc_mode == 2;
	allow_manual = allow_grid_charge || allow_pv_charge || allow_discharge || allow_fc_charge || allow_fc_discharge;
		
	enable('batt_dispatch_choice', en_batt);
	enable('batt_dispatch_auto_can_gridcharge', en_batt && (${batt_dispatch_choice_ui} != 3 && ${batt_dispatch_choice} != 3));
	enable('batt_dispatch_auto_can_charge', en_batt && ${batt_dispatch_choice_ui} != 3);
	enable('batt_dispatch_auto_can_fuelcellcharge', en_batt && ${batt_dispatch_choice_ui} != 3);
	
	enable_manual_inputs(allow_manual, allow_pv_charge, allow_fc_charge, allow_grid_charge, allow_discharge, allow_fc_discharge);
	toggle_discharge_percent();
	toggle_gridcharge_percent();
	toggle_fc_discharge_percent();
//	toggle_custom_dispatch();	

	if (behind_the_meter()){
		toggle_dispatch_target();
		enable_auto_inputs_btm(mode == 4); // Price signal dispatch
	}
}
